20  // Returns Statistics for the given TimeSeries, or None() if the

22  // TODO(dhamon): Consider adding a histogram abstraction for better

24  static Option<Statistics<T> > from(const TimeSeries<T>& timeseries)
25  {
26  std::vector<typename TimeSeries<T>::Value> values_ = timeseries.get();
27 
28  // We need at least 2 values to compute aggregates.
29  if (values_.size() < 2) {
30  return None();
31  }
32

57  }

73  // Returns the requested percentile from the sorted values.
74  // Note that we need at least two values to compute percentiles!
75  // TODO(dhamon): Use a 'Percentage' abstraction.
76  static T percentile(const std::vector<T>& values, double percentile)
77  {
78  CHECK_GE(values.size(), 2u);
79 
80  if (percentile <= 0.0) {
81  return values.front();
82  }
83 
84  if (percentile >= 1.0) {
85  return values.back();
86  }
87 
88  // Use linear interpolation.
89  const double position = percentile * (values.size() - 1);
90  const size_t index = static_cast<size_t>(floor(position));
91  const double delta = position - index;
92 
93  CHECK_GE(index, 0u);
94  CHECK_LT(index, values.size() - 1);
95 
96  return values[index] + delta * (values[index + 1] - values[index]);
97  }
