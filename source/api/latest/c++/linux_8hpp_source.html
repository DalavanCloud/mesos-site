39 inline Result<Process> process(pid_t pid)
40 {
41  // Page size, used for memory accounting.
42  // NOTE: This is more portable than using getpagesize().
43  static const long pageSize = sysconf(_SC_PAGESIZE);
44  if (pageSize <= 0) {
45  return Error("Failed to get sysconf(_SC_PAGESIZE)");
46  }
47 
48  // Number of clock ticks per second, used for cpu accounting.
49  static const long ticks = sysconf(_SC_CLK_TCK);
50  if (ticks <= 0) {
51  return Error("Failed to get sysconf(_SC_CLK_TCK)");
52  }
53 
54  const Result<proc::ProcessStatus>& status = proc::status(pid);
55 
56  if (status.isError()) {
57  return Error(status.error());
58  }
59 
60  if (status.isNone()) {
61  return None();
62  }
63 
64  // There are known bugs with invalid utime / stime values coming
65  // from /proc/<pid>/stat on some Linux systems.
66  // See the following thread for details:
67  // http://mail-archives.apache.org/mod_mbox/incubator-mesos-dev/
68  // 201307.mbox/%3CCA+2n2er-Nemh0CsKLbHRkaHd=YCrNt17NLUPM2=TtEfsKOw4
69  // Rg@mail.gmail.com%3E
70  // These are similar reports:
71  // http://lkml.indiana.edu/hypermail/linux/kernel/1207.1/01388.html
72  // https://bugs.launchpad.net/ubuntu/+source/linux/+bug/1023214
73  Try<Duration> utime = Duration::create(status.get().utime / (double) ticks);
74  Try<Duration> stime = Duration::create(status.get().stime / (double) ticks);
75 
76  // The command line from 'status.get().comm' is only "arg0" from
77  // "argv" (i.e., the canonical executable name). To get the entire
78  // command line we grab '/proc/[pid]/cmdline'.
79  Result<std::string> cmdline = proc::cmdline(pid);
80 
81  return Process(status.get().pid,
82  status.get().ppid,
83  status.get().pgrp,
84  status.get().session,
85  Bytes(status.get().rss * pageSize),
86  utime.isSome() ? utime.get() : Option<Duration>::none(),
87  stime.isSome() ? stime.get() : Option<Duration>::none(),
88  cmdline.isSome() ? cmdline.get() : status.get().comm,
89  status.get().state == 'Z');
90 }
