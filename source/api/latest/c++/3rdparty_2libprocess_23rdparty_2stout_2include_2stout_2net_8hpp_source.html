64 // Returns the HTTP response code resulting from attempting to download the
65 // specified HTTP or FTP URL into a file at the specified path.
66 inline Try<int> download(const std::string& url, const std::string& path)
67 {

69  path, O_CREAT | O_WRONLY, S_IRUSR | S_IWUSR | S_IRGRP | S_IRWXO);
70 
71  if (fd.isError()) {
72  return Error(fd.error());
73  }
74 
75  curl_global_init(CURL_GLOBAL_ALL);
76  CURL* curl = curl_easy_init();
77 
78  if (curl == NULL) {
79  curl_easy_cleanup(curl);
80  os::close(fd.get());
81  return Error("Failed to initialize libcurl");
82  }
83 
84  curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
85  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, NULL);
86  curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, true);
87 
88  FILE* file = fdopen(fd.get(), "w");
89  if (file == NULL) {
90  return ErrnoError("Failed to open file handle of '" + path + "'");
91  }
92  curl_easy_setopt(curl, CURLOPT_WRITEDATA, file);
93 
94  CURLcode curlErrorCode = curl_easy_perform(curl);
95  if (curlErrorCode != 0) {
96  curl_easy_cleanup(curl);
97  fclose(file);
98  return Error(curl_easy_strerror(curlErrorCode));
99  }
100 
101  long code;
102  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &code);
103  curl_easy_cleanup(curl);
104 
105  if (fclose(file) != 0) {
106  return ErrnoError("Failed to close file handle of '" + path + "'");
107  }
108 
109  return Try<int>::some(code);
110 }

163 // Represents a MAC address. A MAC address is a 48-bit unique
164 // identifier assigned to a network interface for communications on
165 // the physical network segment. We use a byte array (in transmission
166 // order) to represent a MAC address. For example, for a MAC address
167 // 01:23:34:67:89:ab, the format is shown as follows:
168 //
169 // MSB LSB
170 // | |
171 // v v
172 // +--------+--------+--------+--------+--------+--------+
173 // |bytes[0]|bytes[1]|bytes[2]|bytes[3]|bytes[4]|bytes[5]|
174 // +--------+--------+--------+--------+--------+--------+
175 //
176 // 01 : 23 : 45 : 67 : 89 : ab
177 class MAC
178 {
179 public:
180  // Constructs a MAC address from a byte array.
181  explicit MAC(const uint8_t (&_bytes)[6])
182  {
183  for (size_t i = 0; i < 6; i++) {
184  bytes[i] = _bytes[i];
185  }
186  }
187 
188  // Returns the byte at the given index. For example, for a MAC
189  // address 01:23:45:67:89:ab, mac[0] = 01, mac[1] = 23 and etc.
190  uint8_t operator [] (size_t index) const
191  {
192  if (index >= 6) {
193  ABORT("Invalid index specified in MAC::operator []\n");
194  }
195 
196  return bytes[index];
197  }
198 
199  bool operator == (const MAC& that) const
200  {
201  for (size_t i = 0; i < 6; i++) {
202  if (bytes[i] != that.bytes[i]) {
203  return false;
204  }
205  }
206  return true;
207  }
208 
209  bool operator != (const MAC& that) const
210  {
211  return !operator == (that);
212  }
213 
214 private:
215  // Byte array of this MAC address (in transmission order).
216  uint8_t bytes[6];
217 };

242 // Returns the MAC address of a given link device. The link device is
243 // specified using its name (e.g., eth0). Returns error if the link
244 // device is not found. Returns none if the link device is found, but
245 // does not have a MAC address (e.g., loopback).
246 inline Result<MAC> mac(const std::string& name)
247 {
248 #if !defined(__linux__) && !defined(__APPLE__)
249  return Error("Not implemented");
250 #else
251  struct ifaddrs* ifaddr = NULL;
252  if (getifaddrs(&ifaddr) == -1) {
253  return ErrnoError();
254  }
255 
256  // Indicates whether the link device is found or not.
257  bool found = false;
258 
259  for (struct ifaddrs* ifa = ifaddr; ifa != NULL; ifa = ifa->ifa_next) {
260  if (ifa->ifa_name != NULL && !strcmp(ifa->ifa_name, name.c_str())) {
261  found = true;
262 
263 # if defined(__linux__)
264  if (ifa->ifa_addr != NULL && ifa->ifa_addr->sa_family == AF_PACKET) {
265  struct sockaddr_ll* link = (struct sockaddr_ll*) ifa->ifa_addr;
266 
267  if (link->sll_halen == 6) {
268  struct ether_addr* addr = (struct ether_addr*) link->sll_addr;
269  MAC mac(addr->ether_addr_octet);
270 
271  // Ignore if the address is 0 so that the results are
272  // consistent on both OSX and Linux.
273  if (stringify(mac) == "00:00:00:00:00:00") {
274  continue;
275  }
276 
277  freeifaddrs(ifaddr);
278  return mac;
279  }
280  }
281 # elif defined(__APPLE__)
282  if (ifa->ifa_addr != NULL && ifa->ifa_addr->sa_family == AF_LINK) {
283  struct sockaddr_dl* link = (struct sockaddr_dl*) ifa->ifa_addr;
284 
285  if (link->sdl_type == IFT_ETHER && link->sdl_alen == 6) {
286  struct ether_addr* addr = (struct ether_addr*) LLADDR(link);
287  MAC mac(addr->octet);
288 
289  freeifaddrs(ifaddr);
290  return mac;
291  }
292  }
293 # endif
294  }
295  }
296 
297  freeifaddrs(ifaddr);
298 
299  if (!found) {
300  return Error("Cannot find the link device");
301  }
302 
303  return None();
304 #endif
305 }

308 // Represents an IPv4 address. Besides the actual IP address, we also
309 // store additional information about the address such as the netmask
310 // which defines the subnet.
311 class IP
312 {
313 public:
314  // Creates an IP from the given string that has the dot-decimal
315  // format (with or without subnet prefix). For example:
316  // 10.0.0.1/8
317  // 192.168.1.100/24
318  // 172.158.1.23

321  // Creates an IP from the given IP address (in host order) and the
322  // given netmask (in host order). Returns error if the netmask is
323  // not valid (e.g., not contiguous).
324  static Try<IP> fromAddressNetmask(uint32_t address, uint32_t netmask);
325 
326  // Creates an IP from a 32-bit address (in host order) and a subnet
327  // prefix (within [0,32]). Returns error if the prefix is not valid.
328  static Try<IP> fromAddressPrefix(uint32_t address, size_t prefix);
329 
330  // Constructs an IP with the given IP address (in host order).
331  explicit IP(uint32_t _address) : address_(_address) {}
332 
333  // Returns the IP address (in host order).
334  uint32_t address() const { return address_; }
335 
336  // Returns the net mask (in host order).
337  Option<uint32_t> netmask() const { return netmask_; }
338 
339  // Returns the prefix of the subnet defined by the net mask.
340  Option<size_t> prefix() const
341  {
342  if (netmask_.isNone()) {
343  return None();
344  }
345 
346  uint32_t mask = netmask_.get();
347 
348  size_t value = 0;
349  while (mask != 0) {
350  value += mask & 1;
351  mask >>= 1;
352  }
353 
354  return value;
355  }
356 
357  bool operator == (const IP& that) const
358  {
359  return address_ == that.address_ && netmask_ == that.netmask_;
360  }
361 
362  bool operator != (const IP& that) const
363  {
364  return !operator == (that);
365  }
366 
367 private:
368  // Constructs an IP with the given IP address (in host order) and
369  // the given netmask (in host order).
370  IP(uint32_t _address, uint32_t _netmask)
371  : address_(_address), netmask_(_netmask) {}
372 
373  // IP address (in host order).
374  uint32_t address_;
375 
376  // The optional net mask (in host order).
377  Option<uint32_t> netmask_;
378 };

381 inline Try<IP> IP::fromDotDecimal(const std::string& value)
382 {
383  std::vector<std::string> tokens = strings::split(value, "/");
384 
385  if (tokens.size() > 2) {
386  return Error("More than one '/' detected");
387  }
388 
389  struct in_addr in;
390  if (inet_aton(tokens[0].c_str(), &in) == 0) {
391  return Error("Failed to parse the IP address");
392  }
393 
394  // Parse the subnet prefix if specified.
395  if (tokens.size() == 2) {
396  Try<size_t> prefix = numify<size_t>(tokens[1]);
397  if (prefix.isError()) {
398  return Error("Subnet prefix is not a number");
399  }
400 
401  return fromAddressPrefix(ntohl(in.s_addr), prefix.get());
402  }
403 
404  return IP(ntohl(in.s_addr));
405 }
