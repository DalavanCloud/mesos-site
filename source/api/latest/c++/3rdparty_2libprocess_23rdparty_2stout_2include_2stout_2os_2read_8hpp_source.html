25 // Reads 'size' bytes from a file from its current offset.
26 // If EOF is encountered before reading 'size' bytes then the result
27 // will contain the bytes read and a subsequent read will return None.
28 inline Result<std::string> read(int fd, size_t size)
29 {
30  char* buffer = new char[size];
31  size_t offset = 0;
32 
33  while (offset < size) {
34  ssize_t length = ::read(fd, buffer + offset, size - offset);
35 
36  if (length < 0) {
37  // TODO(bmahler): Handle a non-blocking fd? (EAGAIN, EWOULDBLOCK)
38  if (errno == EINTR) {
39  continue;
40  }
41  ErrnoError error; // Constructed before 'delete' to capture errno.
42  delete[] buffer;
43  return error;
44  } else if (length == 0) {
45  // Reached EOF before expected! Only return as much data as
46  // available or None if we haven't read anything yet.
47  if (offset > 0) {
48  std::string result(buffer, offset);
49  delete[] buffer;
50  return result;
51  }
52  delete[] buffer;
53  return None();
54  }
55 
56  offset += length;
57  }
58 
59  std::string result(buffer, size);
60  delete[] buffer;
61  return result;
62 }

65 // Returns the contents of the file.
66 inline Try<std::string> read(const std::string& path)
67 {
68  FILE* file = fopen(path.c_str(), "r");
69  if (file == NULL) {
70  return ErrnoError("Failed to open file '" + path + "'");
71  }
72 
73  // Initially the 'line' is NULL and length 0, getline() allocates
74  // ('malloc') a buffer for reading the line.
75  // In subsequent iterations, if the buffer is not large enough to
76  // hold the line, getline() resizes it with 'realloc' and updates
77  // 'line' and 'length' as necessary. See:
78  // - http://pubs.opengroup.org/onlinepubs/9699919799/functions/getline.html
79  // - http://man7.org/linux/man-pages/man3/getline.3.html
80  std::string result;
81  char* line = NULL;
82  size_t length = 0;
83  ssize_t read;
84 
85  while ((read = getline(&line, &length, file)) != -1) {
86  result.append(line, read);
87  }
88 
89  // getline() requires the line buffer to be freed by the caller.
90  free(line);
91 
92  if (ferror(file)) {
93  ErrnoError error;
94  // NOTE: We ignore the error from fclose(). This is because
95  // users calling this function are interested in the return value
96  // of read(). Also an unsuccessful fclose() does not affect the
97  // read.
98  fclose(file);
99  return error;
100  }
101 
102  fclose(file);
103  return result;
104 }
