41 // Defines an abstraction for operations that can be applied on the
42 // Registry.
43 // TODO(xujyan): Make Operation generic so that we can apply them
44 // against a generic "batch operation applier" abstraction, see TODO
45 // below for more details.
46 class Operation : public process::Promise<bool>
47 {
48 public:
49  Operation() : success(false) {}
50  virtual ~Operation() {}
51 
52  // Attempts to invoke the operation on 'registry' (and the
53  // accumulators, in this case 'slaveIDs').
54  // Returns whether the operation mutates 'registry', or an error if
55  // the operation cannot be applied successfully.
56  Try<bool> operator () (
57  Registry* registry,
58  hashset<SlaveID>* slaveIDs,
59  bool strict)
60  {
61  const Try<bool>& result = perform(registry, slaveIDs, strict);
62 
63  success = !result.isError();
64 
65  return result;
66  }
67 
68  // Sets the promise based on whether the operation was successful.
69  bool set() { return process::Promise<bool>::set(success); }
70 
71 protected:
72  virtual Try<bool> perform(
73  Registry* registry,
74  hashset<SlaveID>* slaveIDs,
75  bool strict) = 0;
76 
77 private:
78  bool success;
79 };
