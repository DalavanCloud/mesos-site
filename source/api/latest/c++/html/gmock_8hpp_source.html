158 // A definition of a libprocess filter to enable waiting for events
159 // (such as messages or dispatches) via in tests. This is not meant to
160 // be used directly by tests; tests should use macros like
161 // FUTURE_MESSAGE and FUTURE_DISPATCH instead.
162 class TestsFilter : public Filter
163 {
164 public:
165  TestsFilter()
166  {
167  // We use a recursive mutex here in the event that satisfying the
168  // future created in FutureMessage or FutureDispatch via the
169  // FutureArgField or FutureSatisfy actions invokes callbacks (from
170  // Future::then or Future::onAny, etc) that themselves invoke
171  // FutureDispatch or FutureMessage.
172  pthread_mutexattr_t attr;
173  pthread_mutexattr_init(&attr);
174  pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);
175  pthread_mutex_init(&mutex, &attr);
176  pthread_mutexattr_destroy(&attr);
177  }
178 
179  virtual bool filter(const MessageEvent& event) { return handle(event); }
180  virtual bool filter(const DispatchEvent& event) { return handle(event); }
181  virtual bool filter(const HttpEvent& event) { return handle(event); }
182  virtual bool filter(const ExitedEvent& event) { return handle(event); }
183 
184  template <typename T>
185  bool handle(const T& t)
186  {
187  pthread_mutex_lock(&mutex);
188  bool drop = mock.filter(t);
189  pthread_mutex_unlock(&mutex);
190  return drop;
191  }
192 
193  MockFilter mock;
194  pthread_mutex_t mutex;;
195 };
