354 // For use with a MockScheduler, for example:
355 // EXPECT_CALL(sched, resourceOffers(_, _))
356 // .WillOnce(LaunchTasks(EXECUTOR, TASKS, CPUS, MEM, ROLE));
357 // Launches up to TASKS no-op tasks, if possible,
358 // each with CPUS cpus and MEM memory and EXECUTOR executor.
359 ACTION_P5(LaunchTasks, executor, tasks, cpus, mem, role)
360 {
361  SchedulerDriver* driver = arg0;
362  std::vector<Offer> offers = arg1;
363  int numTasks = tasks;
364 
365  int launched = 0;
366  for (size_t i = 0; i < offers.size(); i++) {
367  const Offer& offer = offers[i];
368 
369  const Resources TASK_RESOURCES = Resources::parse(
370  "cpus:" + stringify(cpus) + ";mem:" + stringify(mem)).get();
371 
372  int nextTaskId = 0;
373  std::vector<TaskInfo> tasks;
374  Resources remaining = offer.resources();
375 
376  while (TASK_RESOURCES <= remaining.flatten() && launched < numTasks) {
377  TaskInfo task;
378  task.set_name("TestTask");
379  task.mutable_task_id()->set_value(stringify(nextTaskId++));
380  task.mutable_slave_id()->MergeFrom(offer.slave_id());
381  task.mutable_executor()->MergeFrom(executor);
382 
383  Option<Resources> resources = remaining.find(TASK_RESOURCES, role);
384  CHECK_SOME(resources);
385  task.mutable_resources()->MergeFrom(resources.get());
386  remaining -= resources.get();
387 
388  tasks.push_back(task);
389  launched++;
390  }
391 
392  driver->launchTasks(offer.id(), tasks);
393  }
394 }
