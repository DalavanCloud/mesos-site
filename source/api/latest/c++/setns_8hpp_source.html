45 // Returns all the supported namespaces by the kernel.
46 inline std::set<std::string> namespaces()
47 {
48  std::set<std::string> result;
49  Try<std::list<std::string> > entries = os::ls("/proc/self/ns");
50  if (entries.isSome()) {
51  foreach (const std::string& entry, entries.get()) {
52  result.insert(entry);
53  }
54  }
55  return result;
56 }

109 // Re-associate the calling process with the specified namespace. The
110 // path refers to one of the corresponding namespace entries in the
111 // /proc/[pid]/ns/ directory (or bind mounted elsewhere). We do not
112 // allow a process with multiple threads to call this function because
113 // it will lead to some weird situations where different threads of a
114 // process are in different namespaces.
115 inline Try<Nothing> setns(const std::string& path, const std::string& ns)
116 {
117  // Return error if there're multiple threads in the calling process.
118  Try<std::set<pid_t> > threads = proc::threads(::getpid());
119  if (threads.isError()) {
120  return Error(
121  "Failed to get the threads of the current process: " +
122  threads.error());
123  } else if (threads.get().size() > 1) {
124  return Error("Multiple threads exist in the current process");
125  }
126 
127  if (os::namespaces().count(ns) == 0) {
128  return Error("Namespace '" + ns + "' is not supported");
129  }
130 
131  // Currently, we don't support pid namespace as its semantics is
132  // different from other namespaces (instead of re-associating the
133  // calling thread, it re-associates the *children* of the calling
134  // thread with the specified namespace).
135  if (ns == "pid") {
136  return Error("Pid namespace is not supported");
137  }
138 
139 #ifdef O_CLOEXEC
140  Try<int> fd = os::open(path, O_RDONLY | O_CLOEXEC);
141 #else
142  Try<int> fd = os::open(path, O_RDONLY);
143 #endif
144 
145  if (fd.isError()) {
146  return Error("Failed to open '" + path + "': " + fd.error());
147  }
148 
149 #ifndef O_CLOEXEC
150  Try<Nothing> cloexec = os::cloexec(fd.get());
151  if (cloexec.isError()) {
152  os::close(fd.get());
153  return Error("Failed to cloexec: " + cloexec.error());
154  }
155 #endif
156 
157  Try<int> nstype = os::nstype(ns);
158  if (nstype.isError()) {
159  return Error(nstype.error());
160  }
161 
162 #ifdef SYS_setns
163  int ret = ::syscall(SYS_setns, fd.get(), nstype.get());
164 #elif __x86_64__
165  // A workaround for those hosts that have an old glibc (older than
166  // 2.14) but have a new kernel. The magic number '308' here is the
167  // syscall number for 'setns' on x86_64 architecture.
168  int ret = ::syscall(308, fd.get(), nstype.get());
169 #else
170 #error "setns is not available"
171 #endif
172 
173  if (ret == -1) {
174  // Save the errno as it might be overwritten by 'os::close' below.
175  ErrnoError error;
176  os::close(fd.get());
177  return error;
178  }
179 
180  os::close(fd.get());
181  return Nothing();
182 }

185 // Re-associate the calling process with the specified namespace. The
186 // pid specifies the process whose namespace we will associate.
187 inline Try<Nothing> setns(pid_t pid, const std::string& ns)
188 {
189  if (!os::exists(pid)) {
190  return Error("Pid " + stringify(pid) + " does not exist");
191  }
192 
193  std::string path = path::join("/proc", stringify(pid), "ns", ns);
194  if (!os::exists(path)) {
195  return Error("Namespace '" + ns + "' is not supported");
196  }
197 
198  return os::setns(path, ns);
199 }
