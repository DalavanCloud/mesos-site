36 // A basic IsolatorProcess that keeps track of the pid but doesn't do any
37 // resource isolation. Subclasses must implement usage() for their appropriate
38 // resource(s).
39 class PosixIsolatorProcess : public IsolatorProcess
40 {
41 public:
42  virtual process::Future<Nothing> recover(
43  const std::list<state::RunState>& state)
44  {
45  foreach (const state::RunState& run, state) {
46  if (!run.id.isSome()) {
47  return process::Failure("ContainerID is required to recover");
48  }
49 
50  if (!run.forkedPid.isSome()) {
51  return process::Failure("Executor pid is required to recover");
52  }
53 
54  // This should (almost) never occur: see comment in
55  // PosixLauncher::recover().
56  if (pids.contains(run.id.get())) {
57  return process::Failure("Container already recovered");
58  }
59 
60  pids.put(run.id.get(), run.forkedPid.get());
61 
62  process::Owned<process::Promise<Limitation> > promise(
63  new process::Promise<Limitation>());
64  promises.put(run.id.get(), promise);
65  }
66 
67  return Nothing();
68  }
69 
70  virtual process::Future<Option<CommandInfo> > prepare(
71  const ContainerID& containerId,
72  const ExecutorInfo& executorInfo)
73  {
74  if (promises.contains(containerId)) {
75  return process::Failure("Container " + stringify(containerId) +
76  " has already been prepared");
77  }
78 
79  process::Owned<process::Promise<Limitation> > promise(
80  new process::Promise<Limitation>());
81  promises.put(containerId, promise);
82 
83  return None();
84  }
85 
86  virtual process::Future<Nothing> isolate(
87  const ContainerID& containerId,
88  pid_t pid)
89  {
90  if (!promises.contains(containerId)) {
91  return process::Failure("Unknown container: " + stringify(containerId));
92  }
93 
94  pids.put(containerId, pid);
95 
96  return Nothing();
97  }
98 
99  virtual process::Future<Limitation> watch(
100  const ContainerID& containerId)
101  {
102  if (!promises.contains(containerId)) {
103  return process::Failure("Unknown container: " + stringify(containerId));
104  }
105 
106  return promises[containerId]->future();
107  }
108 
109  virtual process::Future<Nothing> update(
110  const ContainerID& containerId,
111  const Resources& resources)
112  {
113  if (!promises.contains(containerId)) {
114  return process::Failure("Unknown container: " + stringify(containerId));
115  }
116 
117  // No resources are actually isolated so nothing to do.
118  return Nothing();
119  }
120 
121  virtual process::Future<Nothing> cleanup(const ContainerID& containerId)
122  {
123  if (!promises.contains(containerId)) {
124  return process::Failure("Unknown container: " + stringify(containerId));
125  }
126 
127  // TODO(idownes): We should discard the container's promise here to signal
128  // to anyone that holds the future from watch().
129  promises.erase(containerId);
130 
131  pids.erase(containerId);
132 
133  return Nothing();
134  }
135 
136 protected:
137  hashmap<ContainerID, pid_t> pids;
138  hashmap<ContainerID,
139  process::Owned<process::Promise<Limitation> > > promises;
140 };
