18 // Provides an abstraction that serializes the execution of a sequence
19 // of callbacks.
20 class Sequence
21 {
22 public:
23  Sequence();
24  ~Sequence();
25 
26  // Registers a callback that will be invoked when all the futures
27  // returned by the previously registered callbacks are in
28  // non-pending status (i.e., ready, discarded or failed). Due to
29  // these semantics, we should avoid registering a callback which
30  // returns a future that could be in non-pending status for a long
31  // time because it will prevent all the subsequent callbacks from
32  // being invoked. This is analogous to the requirement that we want
33  // to avoid invoking blocking function in a libprocess handler. A
34  // user is allowed to cancel a registered callback by discarding the
35  // returned future. Other callbacks in this sequence will NOT be
36  // affected. The subsequent callbacks will not be invoked until the
37  // future is actually DISCARDED.
38  template <typename T>
39  Future<T> add(const lambda::function<Future<T>(void)>& callback);
40 
41 private:
42  // Not copyable, not assignable.
43  Sequence(const Sequence&);
44  Sequence& operator = (const Sequence&);
45 
46  SequenceProcess* process;
47 };

50 class SequenceProcess : public Process<SequenceProcess>
51 {
52 public:
53  SequenceProcess() : last(Nothing()) {}
54 
55  template <typename T>
56  Future<T> add(const lambda::function<Future<T>(void)>& callback)
57  {
58  // This is the future that is used to notify the next callback
59  // (denoted by 'N' in the following graph).
60  Owned<Promise<Nothing> > notifier(new Promise<Nothing>());
61 
62  // This is the future that will be returned to the user (denoted
63  // by 'F' in the following graph).
64  Owned<Promise<T> > promise(new Promise<T>());
65 
66  // We use a graph to show how we hook these futures. Each box in
67  // the graph represents a future. As mentioned above, 'F' denotes
68  // a future that will be returned to the user, and 'N' denotes a
69  // future that is used for notifying the next callback. Each arrow
70  // represents a "notification" relation. We will explain in detail
71  // what "notification" means in the following.
72  //
73  // 'last' 'last' 'last'
74  // | | |
75  // v v v
76  // +---+ +---+ +---+ +---+ +---+ +---+
77  // | N | | N |--+ | N | | N |--+ | N |--+ | N |
78  // +---+ +---+ | +---+ +---+ | +---+ | +---+
79  // ==> | ^ ==> | ^ | ^
80  // | | | | | |
81  // | +---+ | +---+ | +---+
82  // +-->| F | +-->| F | +-->| F |
83  // +---+ +---+ +---+
84  //
85  // Initial => Added one callback => Added two callbacks
86 
87  // Setup the "notification" from 'F' to 'N' so that when a
88  // callback is done, signal the notifier ('N').
89  promise->future().onAny(lambda::bind(&completed, notifier));
90 
91  // Setup the "notification" from previous 'N' to 'F' so that when
92  // a notifier ('N') is set (indicating the previous callback has
93  // completed), invoke the next callback ('F') in the sequence.
94  last.onAny(lambda::bind(&notified<T>, promise, callback));
95 
96  // In the following, we setup the hooks so that if this sequence
97  // process is being terminated, all pending callbacks will be
98  // discarded. We use weak futures here to avoid cyclic dependencies.
99 
100  // Discard the future associated with this notifier.
101  notifier->future().onDiscard(
102  lambda::bind(
103  &internal::discard<T>,
104  WeakFuture<T>(promise->future())));
105 
106  // Discard the notifier associated with the previous future.
107  notifier->future().onDiscard(
108  lambda::bind(
109  &internal::discard<Nothing>,
110  WeakFuture<Nothing>(last)));
111 
112  // Update the 'last'.
113  last = notifier->future();
114 
115  return promise->future();
116  }
117 
118 protected:
119  virtual void finalize()
120  {
121  last.discard();
122 
123  // TODO(jieyu): Do we need to wait for the future of the last
124  // callback to be in DISCARDED state?
125  }
126 
127 private:
128  // Invoked when a callback is done.
129  static void completed(Owned<Promise<Nothing> > notifier)
130  {
131  notifier->set(Nothing());
132  }
133 
134  // Invoked when a notifier is set.
135  template <typename T>
136  static void notified(
137  Owned<Promise<T> > promise,
138  const lambda::function<Future<T>(void)>& callback)
139  {
140  if (promise->future().hasDiscard()) {
141  // The user has shown the intention to discard this callback
142  // (i.e., by calling future.discard()). As a result, we will
143  // just skip this callback.
144  promise->discard();
145  } else {
146  promise->associate(callback());
147  }
148  }
149 
150  Future<Nothing> last;
151 };
