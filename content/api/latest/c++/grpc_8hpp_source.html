146  template <
147  typename Method,
148  typename Request =
149  typename internal::MethodTraits<Method>::request_type,
150  typename Response =
151  typename internal::MethodTraits<Method>::response_type>
152  Future<Try<Response, StatusError>> call(
153  const Connection& connection,
154  Method&& method,
155  const Request& request)
156  {
157  static_assert(
158  std::is_convertible<Request*, google::protobuf::Message*>::value,
159  "Request must be a protobuf message");
160 
161  synchronized (data->lock) {
162  if (data->terminating) {
163  return Failure("Runtime has been terminated.");
164  }
165 
166  std::shared_ptr<::grpc::ClientContext> context(
167  new ::grpc::ClientContext());
168 
169  // TODO(chhsiao): Allow the caller to specify a timeout.
170  context->set_deadline(
171  std::chrono::system_clock::now() + std::chrono::seconds(5));
172 
173  // Enable the gRPC wait-for-ready semantics by default. See:
174  // https://github.com/grpc/grpc/blob/master/doc/wait-for-ready.md
175  // TODO(chhsiao): Allow the caller to set the option.
176  context->set_wait_for_ready(true);
177 
178  // Create a `Promise` and a callback lambda as a tag and invokes
179  // an asynchronous gRPC call through the `CompletionQueue`
180  // managed by `data`. The `Promise` will be set by the callback
181  // upon server response.
182  std::shared_ptr<Promise<Try<Response, StatusError>>> promise(
183  new Promise<Try<Response, StatusError>>);
184 
185  promise->future().onDiscard([=] { context->TryCancel(); });
186 
187  std::shared_ptr<Response> response(new Response());
188  std::shared_ptr<::grpc::Status> status(new ::grpc::Status());
189 
190  std::shared_ptr<::grpc::ClientAsyncResponseReader<Response>> reader =
191  (typename internal::MethodTraits<Method>::stub_type(
192  connection.channel).*method)(context.get(), request, &data->queue);
193 
194  reader->StartCall();
195  reader->Finish(
196  response.get(),
197  status.get(),
198  new lambda::function<void()>(
199  // NOTE: `context` and `reader` need to be held on in
200  // order to get updates for the ongoing RPC, and thus
201  // are captured here. The lambda itself will later be
202  // retrieved and managed in `Data::loop()`.
203  [context, reader, response, status, promise]() {
204  CHECK(promise->future().isPending());
205  if (promise->future().hasDiscard()) {
206  promise->discard();
207  } else {
208  promise->set(status->ok()
209  ? std::move(*response)
210  : Try<Response, StatusError>::error(std::move(*status)));
211  }
212  }));
213 
214  return promise->future();
215  }
216  }
