119  Future<RpcResult<Response>> call(

123  const Request&,

127  static_assert(
128  std::is_convertible<Request*, google::protobuf::Message*>::value,
129  "Request must be a protobuf message");
130 
131  synchronized (data->lock) {
132  if (data->terminating) {
133  return Failure("Runtime has been terminated.");
134  }
135 
136  std::shared_ptr<::grpc::ClientContext> context(
137  new ::grpc::ClientContext());
138 
139  // TODO(chhsiao): Allow the caller to specify a timeout.
140  context->set_deadline(
141  std::chrono::system_clock::now() + std::chrono::seconds(5));
142 
143  // Enable the gRPC wait-for-ready semantics by default. See:
144  // https://github.com/grpc/grpc/blob/master/doc/wait-for-ready.md
145  // TODO(chhsiao): Allow the caller to set the option.
146  context->set_wait_for_ready(true);
147 
148  // Create a `Promise` and a callback lambda as a tag and invokes
149  // an asynchronous gRPC call through the `CompletionQueue`
150  // managed by `data`. The `Promise` will be set by the callback
151  // upon server response.
152  std::shared_ptr<Promise<RpcResult<Response>>> promise(
153  new Promise<RpcResult<Response>>);
154 
155  promise->future().onDiscard([=] { context->TryCancel(); });
156 
157  std::shared_ptr<Response> response(new Response());
158  std::shared_ptr<::grpc::Status> status(new ::grpc::Status());
159 
160  std::shared_ptr<::grpc::ClientAsyncResponseReader<Response>> reader(
161  (Stub(channel.channel).*rpc)(context.get(), request, &data->queue));
162 
163  reader->Finish(
164  response.get(),
165  status.get(),
166  new lambda::function<void()>(
167  // NOTE: `context` and `reader` need to be held on in
168  // order to get updates for the ongoing RPC, and thus
169  // are captured here. The lambda itself will later be
170  // retrieved and managed in `Data::loop()`.
171  [context, reader, response, status, promise]() {
172  CHECK(promise->future().isPending());
173  if (promise->future().hasDiscard()) {
174  promise->discard();
175  return;
176  }
177 
178  promise->set(RpcResult<Response>(*status, *response));
179  }));
180 
181  return promise->future();
182  }
183  }
