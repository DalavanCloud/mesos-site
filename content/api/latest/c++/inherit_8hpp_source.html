27 // This function enables or disables inheritance for a Windows file handle.
28 //
29 // NOTE: By default, handles on Windows are not inheritable, so this is
30 // primarily used to enable inheritance when passing handles to child processes,
31 // and subsequently disable inheritance.
32 inline Try<Nothing> set_inherit(const os::WindowsFD& fd, const bool inherit)
33 {
34  const BOOL result = ::SetHandleInformation(
35  fd, HANDLE_FLAG_INHERIT, inherit ? HANDLE_FLAG_INHERIT : 0);
36

38  return WindowsError();
39  }
40

42 }
