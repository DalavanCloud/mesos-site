47 // Resource format options to be used with the `convertResourceFormat` function.
48 //
49 // The preconditions of the options are asymmetric, centered around the
50 // "post-reservation-refinement" format. This is mainly due to the fact that
51 // "post-reservation-refinement" format is our canonical representation.
52 // The transformations are generally applied to any of the 3 formats to be
53 // converted to the canonical format, then later converted back as necessary.
54 //
55 // See 'Resource Format' section in `mesos.proto` for more details.
56 enum ResourceFormat
57 {
58  // "post-reservation-refinement" -> "pre-reservation-refinement"
59  //
60  // The `Resource` objects must be in the "post-reservation-refinement" format,
61  // and must not have refined reservations.
62  //
63  // All resources end up with the `Resource.role` and `Resource.reservation`
64  // fields set, and the `Resource.reservations` field unset.
65  //
66  // We convert the resources to the "pre-reservation-refinement" format to
67  // checkpoint resources for example. This enables downgrading to an agent
68  // without a RESERVATION_REFINEMENT caapbility, since the resources will
69  // be checkpointed in a format that the downgraded agent can recover from.
70  PRE_RESERVATION_REFINEMENT,
71 
72  // "pre-reservation-refinement" -> "post-reservation-refinement"
73  // "post-reservation-refinement" -> "post-reservation-refinement"
74  // "endpoint" -> "post-reservation-refinement"
75  //
76  // The `Resource` objects can be in any of the valid resource formats:
77  // "pre-reservation-refinement", "post-reservation-refinement", "endpoint".
78  //
79  // All resources end up with the `Resource.reservations` field set,
80  // and the `Resource.role` and `Resource.reservation` fields unset.
81  //
82  // If the `Resource` objects are already in the "post-reservation-refinement"
83  // format, this is a no-op.
84  //
85  // We convert the resources to the "post-reservation-refinement" format,
86  // for example, when a master receives a message from an agent without
87  // the RESERVATION_REFINEMENT capability. This allows a component
88  // (e.g. master) code to deal with a canonical format to simplify the code.
89  POST_RESERVATION_REFINEMENT,
90 
91  // "post-reservation-refinement" -> "endpoint"
92  //
93  // This is a special case for endpoints, which injects
94  // the "pre-reservation-refinement" format.
95  //
96  // The `Resource` objects must be in the "post-reservation-refinement" format.
97  //
98  // All resources continue to have the `Resource.reservations` field set.
99  // The `Resource` objects without refined reservations end up with the
100  // `Resource.role` and `Resource.reservation` fields set, and the objects
101  // with refined reservations have them unset.
102  //
103  // We inject the resources with the "pre-reservation-refinement" format to
104  // enable backward compatibility with external tooling. If the master has been
105  // upgraded to a version that supports reservation refinement but no refined
106  // reservations have been made, the endpoints will return the data in both new
107  // and old formats to maximize backward compatibility. However, once
108  // a reservation refinement is made to a resource, that resource is only
109  // returned in the new format.
110  ENDPOINT
111 };
