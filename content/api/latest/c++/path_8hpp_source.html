69 inline bool absolute(const std::string& path)
70 {
71 #ifndef __WINDOWS__
72  return strings::startsWith(path, os::PATH_SEPARATOR);
73 #else
74  // NOTE: We do not use `PathIsRelative` Windows utility function
75  // here because it does not support long paths.
76  //
77  // See https://msdn.microsoft.com/en-us/library/windows/desktop/aa365247(v=vs.85).aspx
78  // for details on paths. In short, an absolute path for files on Windows
79  // looks like one of the following:
80  // * "[A-Za-z]:\"
81  // * "[A-Za-z]:/"
82  // * "\\?\..."
83  // * "\\server\..." where "server" is a network host.
84  //
85  // NOLINT(whitespace/line_length)
86 
87  // A uniform naming convention (UNC) name of any format,
88  // always starts with two backslash characters.
89  if (strings::startsWith(path, "\\\\")) {
90  return true;
91  }
92 
93  // A disk designator with a slash, for example "C:\" or "d:/".
94  if (path.length() < 3) {
95  return false;
96  }
97 
98  const char letter = path[0];
99  if (!((letter >= 'A' && letter <= 'Z') ||
100  (letter >= 'a' && letter <= 'z'))) {
101  return false;
102  }
103 
104  std::string colon = path.substr(1, 2);
105  return colon == ":\\" || colon == ":/";
106 #endif // __WINDOWS__
107 }

152  inline std::string basename() const
153  {
154  if (value.empty()) {
155  return std::string(".");
156  }
157 
158  size_t end = value.size() - 1;
159 
160  // Remove trailing slashes.
161  if (value[end] == os::PATH_SEPARATOR) {
162  end = value.find_last_not_of(os::PATH_SEPARATOR, end);
163 
164  // Paths containing only slashes result into "/".
165  if (end == std::string::npos) {
166  return stringify(os::PATH_SEPARATOR);
167  }
168  }
169 
170  // 'start' should point towards the character after the last slash
171  // that is non trailing.
172  size_t start = value.find_last_of(os::PATH_SEPARATOR, end);
173 
174  if (start == std::string::npos) {
175  start = 0;
176  } else {
177  start++;
178  }
179 
180  return value.substr(start, end + 1 - start);
181  }
