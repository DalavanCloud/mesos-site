40 // Retrieves system environment in a `std::map`, ignoring
41 // the current process's environment variables.
42 inline Option<std::map<std::wstring, std::wstring>> get_system_env()
43 {
44  std::map<std::wstring, std::wstring> system_env;
45  wchar_t* env_entry = nullptr;
46 
47  // Get the system environment.
48  // The third parameter (bool) tells the function *not* to inherit
49  // variables from the current process.
50  if (!::CreateEnvironmentBlock((LPVOID*)&env_entry, nullptr, FALSE)) {
51  return None();
52  }
53 
54  // Save the environment block in order to destroy it later.
55  wchar_t* env_block = env_entry;
56 
57  while (*env_entry != L'\0') {
58  // Each environment block contains the environment variables as follows:
59  // Var1=Value1\0
60  // Var2=Value2\0
61  // Var3=Value3\0
62  // ...
63  // VarN=ValueN\0\0
64  // The name of an environment variable cannot include an equal sign (=).
65 
66  // Construct a string from the pointer up to the first '\0',
67  // e.g. "Var1=Value1\0", then split into name and value.
68  std::wstring entry(env_entry);
69  std::wstring::size_type separator = entry.find(L"=");
70  std::wstring var_name(entry.substr(0, separator));
71  std::wstring varVal(entry.substr(separator + 1));
72 
73  // Mesos variables are upper case. Convert system variables to
74  // match the name provided by the scheduler in case of a collision.
75  // This is safe because Windows environment variables are case insensitive.
76  std::transform(
77  var_name.begin(), var_name.end(), var_name.begin(), ::towupper);
78 
79  // The system environment has priority.
80  system_env.insert_or_assign(var_name.data(), varVal.data());
81 
82  // Advance the pointer the length of the entry string plus the '\0'.
83  env_entry += entry.length() + 1;
84  }
85 
86  ::DestroyEnvironmentBlock(env_block);
87 
88  return system_env;
89 }

92 // Creates a null-terminated array of null-terminated strings that will be
93 // passed to `CreateProcessW` as the `lpEnvironment` argument, as described by
94 // MSDN[1]. This array needs to be sorted in alphabetical order, but the `map`
95 // already takes care of that. Note that this function explicitly handles
96 // UTF-16 environments, so it must be used in conjunction with the
97 // `CREATE_UNICODE_ENVIRONMENT` flag.
98 //
99 // NOTE: This function will add the system's environment variables into
100 // the returned string. These variables take precedence over the provided
101 // `env` and are generally necessary in order to launch things on Windows.
102 //
103 // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/ms682425(v=vs.85).aspx
104 inline Option<std::wstring> create_process_env(
105  const Option<std::map<std::string, std::string>>& env)
106 {
107  if (env.isNone() || (env.isSome() && env.get().size() == 0)) {
108  return None();
109  }
110 
111  Option<std::map<std::wstring, std::wstring>> system_env = get_system_env();
112 
113  // The system environment must be non-empty.
114  // No subprocesses will be able to launch if the system environment is blank.
115  CHECK(system_env.isSome() && system_env.get().size() > 0);
116 
117  std::map<std::wstring, std::wstring> combined_env;
118 
119  // Populate the combined environment first with the given environment
120  // converted to UTF-16 for Windows.
121  foreachpair (const std::string& key,
122  const std::string& value,
123  env.get()) {
124  combined_env[wide_stringify(key)] = wide_stringify(value);
125  }
126 
127  // Add the system environment variables, overwriting the previous.
128  foreachpair (const std::wstring& key,
129  const std::wstring& value,
130  system_env.get()) {
131  combined_env[key] = value;
132  }
133 
134  std::wstring env_string;
135  foreachpair (const std::wstring& key,
136  const std::wstring& value,
137  combined_env) {
138  env_string += key + L'=' + value + L'\0';
139  }
140 
141  // Append final null terminating character.
142  env_string.push_back(L'\0');
143  return env_string;
144 }

147 // Concatenates multiple command-line arguments and escapes the values.
148 // NOTE: This is necessary even when using Windows APIs that "appear"
149 // to take arguments as a list, because those APIs will themselves
150 // concatenate command-line arguments *without* escaping them.
151 //
152 // This function escapes arguments with the following rules:
153 // 1) Any argument with a space, tab, newline, vertical tab,
154 // or double-quote must be surrounded in double-quotes.
155 // 2) Backslashes at the very end of an argument must be escaped.
156 // 3) Backslashes that precede a double-quote must be escaped.
157 // The double-quote must also be escaped.
158 //
159 // NOTE: The below algorithm is adapted from Daniel Colascione's public domain
160 // algorithm for quoting command line arguments on Windows for `CreateProcess`.
161 //
162 // https://blogs.msdn.microsoft.com/twistylittlepassagesallalike/2011/04/23/everyone-quotes-command-line-arguments-the-wrong-way/
163 // NOLINT(whitespace/line_length)
164 inline std::wstring stringify_args(const std::vector<std::string>& argv)
165 {
166  std::wstring command;
167  for (auto argit = argv.cbegin(); argit != argv.cend(); ++argit) {
168  std::wstring arg = wide_stringify(*argit);
169  // Don't quote empty arguments or those without troublesome characters.
170  if (!arg.empty() && arg.find_first_of(L" \t\n\v\"") == arg.npos) {
171  command.append(arg);
172  } else {
173  // Beginning double quotation mark.
174  command.push_back(L'"');
175  for (auto it = arg.cbegin(); it != arg.cend(); ++it) {
176  // Count existent backslashes in argument.
177  unsigned int backslashes = 0;
178  while (it != arg.cend() && *it == L'\\') {
179  ++it;
180  ++backslashes;
181  }
182 
183  if (it == arg.cend()) {
184  // Escape all backslashes, but let the terminating double quotation
185  // mark we add below be interpreted as a metacharacter.
186  command.append(backslashes * 2, L'\\');
187  break;
188  } else if (*it == L'"') {
189  // Escape all backslashes and the following double quotation mark.
190  command.append(backslashes * 2 + 1, L'\\');
191  command.push_back(*it);
192  } else {
193  // Backslashes aren't special here.
194  command.append(backslashes, L'\\');
195  command.push_back(*it);
196  }
197  }
198 
199  // Terminating double quotation mark.
200  command.push_back(L'"');
201  }
202  // Space separate arguments (but don't append at end).
203  if (argit != argv.cend() - 1) {
204  command.push_back(L' ');
205  }
206  }
207  // Append final null terminating character.
208  command.push_back(L'\0');
209  return command;
210 }

218 // Provides an interface for creating a child process on Windows.
219 //
220 // The `command` argument is given for compatibility, and is ignored. This is
221 // because the `CreateProcess` will use `argv[0]` as the command to be executed,
222 // and will perform a `PATH` lookup. If `command` were to be used instead,
223 // `CreateProcess` would require an absolute path.
224 //
225 // If `create_suspended` is `true`, the process will not be started, and the
226 // caller must use `ResumeThread` to start the process.
227 //
228 // The caller can specify explicit `stdin`, `stdout`, and `stderr` handles,
229 // in that order, for the process via the `pipes` argument.
230 //
231 // The return value is a `ProcessData` struct, with the process and thread
232 // handles each saved in a `SharedHandle`, ensuring they are closed when struct
233 // goes out of scope.
234 inline Try<ProcessData> create_process(
235  const std::string& command,
236  const std::vector<std::string>& argv,
237  const Option<std::map<std::string, std::string>>& environment,
238  const bool create_suspended = false,
239  const Option<std::tuple<int_fd, int_fd, int_fd>> pipes = None())
240 {
241  // TODO(andschwa): Assert that `command` and `argv[0]` are the same.
242  std::wstring arg_string = stringify_args(argv);
243  std::vector<wchar_t> arg_buffer(arg_string.begin(), arg_string.end());
244  arg_buffer.push_back(L'\0');
245 
246  // Create the process with a Unicode environment.
247  DWORD creation_flags = CREATE_UNICODE_ENVIRONMENT;
248  if (create_suspended) {
249  creation_flags |= CREATE_SUSPENDED;
250  }
251 
252  // Construct the environment that will be passed to `::CreateProcessW`.
253  Option<std::wstring> env_string = create_process_env(environment);
254  std::vector<wchar_t> env_buffer;
255  if (env_string.isSome()) {
256  // This string contains the necessary null characters.
257  env_buffer.assign(env_string.get().begin(), env_string.get().end());
258  }
259 
260  wchar_t* process_env = env_buffer.empty() ? nullptr : env_buffer.data();
261 
262  PROCESS_INFORMATION process_info;
263  memset(&process_info, 0, sizeof(PROCESS_INFORMATION));
264 
265  STARTUPINFOW startup_info;
266  memset(&startup_info, 0, sizeof(STARTUPINFOW));
267  startup_info.cb = sizeof(STARTUPINFOW);
268 
269  // Hook up the stdin/out/err pipes and use the `STARTF_USESTDHANDLES`
270  // flag to instruct the child to use them [1].
271  // A more user-friendly example can be found in [2].
272  //
273  // [1] https://msdn.microsoft.com/en-us/library/windows/desktop/ms686331(v=vs.85).aspx
274  // [2] https://msdn.microsoft.com/en-us/library/windows/desktop/ms682499(v=vs.85).aspx
275  if (pipes.isSome()) {
276  startup_info.hStdInput = std::get<0>(pipes.get());
277  startup_info.hStdOutput = std::get<1>(pipes.get());
278  startup_info.hStdError = std::get<2>(pipes.get());
279  startup_info.dwFlags |= STARTF_USESTDHANDLES;
280  }
281 
282  BOOL create_process_result = ::CreateProcessW(
283  // This is replaced by the first token of `arg_buffer` string.
284  static_cast<LPCWSTR>(nullptr),
285  static_cast<LPWSTR>(arg_buffer.data()),
286  static_cast<LPSECURITY_ATTRIBUTES>(nullptr),
287  static_cast<LPSECURITY_ATTRIBUTES>(nullptr),
288  TRUE, // Inherited parent process handles.
289  creation_flags,
290  static_cast<LPVOID>(process_env),
291  static_cast<LPCWSTR>(nullptr), // Inherited working directory.
292  &startup_info,
293  &process_info);
294 
295  if (!create_process_result) {
296  return WindowsError(
297  "Failed to call `CreateProcess`: " + stringify(arg_string));
298  }
299 
300  return ProcessData{
301  SharedHandle{process_info.hProcess, ::CloseHandle},
302  SharedHandle{process_info.hThread, ::CloseHandle},
303  static_cast<pid_t>(process_info.dwProcessId)};
304 }

332 // Executes a command by calling "<command> <arguments...>", and
333 // returns after the command has been completed. Returns process exit code if
334 // succeeds, and -1 on error.
335 inline int spawn(
336  const std::string& command,
337  const std::vector<std::string>& arguments,
338  const Option<std::map<std::string, std::string>>& environment = None())
339 {
340  Try<::internal::windows::ProcessData> process_data =
341  ::internal::windows::create_process(command, arguments, environment);
342 
343  if (process_data.isError()) {
344  LOG(WARNING) << process_data.error();
345  return -1;
346  }
347 
348  // Wait for the process synchronously.
349  ::WaitForSingleObject(
350  process_data.get().process_handle.get_handle(), INFINITE);
351 
352  DWORD status;
353  if (!::GetExitCodeProcess(
354  process_data.get().process_handle.get_handle(),
355  &status)) {
356  LOG(WARNING) << "Failed to `GetExitCodeProcess`: " << command;
357  return -1;
358  }
359 
360  // Return the child exit code.
361  return static_cast<int>(status);
362 }
