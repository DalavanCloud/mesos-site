25 // Creates a temporary directory using the specified path
26 // template. The template may be any path with _6_ `Xs' appended to
27 // it, for example /tmp/temp.XXXXXX. The trailing `Xs' are replaced
28 // with a unique alphanumeric combination.
29 inline Try<std::string> mkdtemp(const std::string& path = "/tmp/XXXXXX")
30 {
31  char* temp = new char[path.size() + 1];
32  if (::mkdtemp(::strcpy(temp, path.c_str())) != NULL) {
33  std::string result(temp);
34  delete[] temp;
35  return result;
36  } else {
37  delete[] temp;
38  return ErrnoError();
39  }
40 }
