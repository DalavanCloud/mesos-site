30 // Docker expects the image to be specified on the command line as:
31 // [REGISTRY_HOST[:REGISTRY_PORT]/]REPOSITORY[:TAG|@TYPE:DIGEST]
32 //
33 // This format is inherently ambiguous when dealing with repository
34 // names that include forward slashes. To disambiguate, the docker
35 // code looks for '.', or ':', or 'localhost' to decide if the
36 // first component is a registry or a respository name. For more
37 // detail, drill into the implementation of docker pull.
38 //
39 // TODO(bmahler): We currently store the digest as a tag, does
40 // that makes sense?
41 //
42 // TODO(bmahler): Validate based on docker's validation logic
43 // and return a Try here.
44 //
45 // TODO(jieyu): Remove this in favor of using 'spec::parseImageName'.
46 inline Image::Name parseImageName(std::string s)
47 {
48  Image::Name name;
49 
50  // Extract the digest.
51  if (strings::contains(s, "@")) {
52  std::vector<std::string> split = strings::split(s, "@");
53 
54  s = split[0];
55  name.set_tag(split[1]);
56  }
57 
58  // Remove the tag. We need to watch out for a
59  // host:port registry, which also contains ':'.
60  if (strings::contains(s, ":")) {
61  std::vector<std::string> split = strings::split(s, ":");
62 
63  // The tag must be the last component. If a slash is
64  // present there is a registry port and no tag.
65  if (!strings::contains(split.back(), "/")) {
66  name.set_tag(split.back());
67  split.pop_back();
68 
69  s = strings::join(":", split);
70  }
71  }
72 
73  // Default to the 'latest' tag when omitted.
74  if (name.tag().empty()) {
75  name.set_tag("latest");
76  }
77 
78  // Extract the registry and repository. The first component can
79  // either be the registry, or the first part of the repository!
80  // We resolve this ambiguity using the same hacks used in the
81  // docker code ('.', ':', 'localhost' indicate a registry).
82  std::vector<std::string> split = strings::split(s, "/", 2);
83 
84  if (split.size() == 1) {
85  name.set_repository(s);
86  } else if (strings::contains(split[0], ".") ||
87  strings::contains(split[0], ":") ||
88  split[0] == "localhost") {
89  name.set_registry(split[0]);
90  name.set_repository(split[1]);
91  } else {
92  name.set_repository(s);
93  }
94 
95  return name;
96 }
