86 class HealthCheckerProcess : public ProtobufProcess<HealthCheckerProcess>
87 {
88 public:
89  HealthCheckerProcess(
90  const HealthCheck& _check,
91  const UPID& _executor,
92  const TaskID& _taskID)
93  : check(_check),
94  initializing(true),
95  executor(_executor),
96  taskID(_taskID),
97  consecutiveFailures(0) {}
98 
99  virtual ~HealthCheckerProcess() {}
100 
101  process::Future<Nothing> healthCheck()
102  {
103  VLOG(2) << "Health checks starting in "
104  << Seconds(check.delay_seconds()) << ", grace period "
105  << Seconds(check.grace_period_seconds());
106 
107  startTime = process::Clock::now();
108 
109  delay(Seconds(check.delay_seconds()), self(), &Self::_healthCheck);
110  return promise.future();
111  }
112 
113 private:
114  void failure(const string& message)
115  {
116  if (check.grace_period_seconds() > 0 &&
117  (process::Clock::now() - startTime).secs() <=
118  check.grace_period_seconds()) {
119  LOG(INFO) << "Ignoring failure as health check still in grace period";
120  reschedule();
121  return;
122  }
123 
124  consecutiveFailures++;
125  VLOG(1) << "#" << consecutiveFailures << " check failed: " << message;
126 
127  bool killTask = consecutiveFailures >= check.consecutive_failures();
128 
129  TaskHealthStatus taskHealthStatus;
130  taskHealthStatus.set_healthy(false);
131  taskHealthStatus.set_consecutive_failures(consecutiveFailures);
132  taskHealthStatus.set_kill_task(killTask);
133  taskHealthStatus.mutable_task_id()->CopyFrom(taskID);
134  send(executor, taskHealthStatus);
135 
136  if (killTask) {
137  // This is a hack to ensure the message is sent to the
138  // executor before we exit the process. Without this,
139  // we may exit before libprocess has sent the data over
140  // the socket. See MESOS-4111.
141  os::sleep(Seconds(1));
142  promise.fail(message);
143  } else {
144  reschedule();
145  }
146  }
147 
148  void success()
149  {
150  VLOG(1) << "Check passed";
151 
152  // Send a healthy status update on the first success,
153  // and on the first success following failure(s).
154  if (initializing || consecutiveFailures > 0) {
155  TaskHealthStatus taskHealthStatus;
156  taskHealthStatus.set_healthy(true);
157  taskHealthStatus.mutable_task_id()->CopyFrom(taskID);
158  send(executor, taskHealthStatus);
159  initializing = false;
160  }
161  consecutiveFailures = 0;
162  reschedule();
163  }
164 
165  void _healthCheck()
166  {
167  if (check.has_http()) {
168  promise.fail("HTTP health check is not supported");
169  return;
170  }
171 
172  if (!check.has_command()) {
173  promise.fail("No check found in health check");
174  return;
175  }
176 
177  const CommandInfo& command = check.command();
178 
179  map<string, string> environment = os::environment();
180 
181  foreach (const Environment::Variable& variable,
182  command.environment().variables()) {
183  environment[variable.name()] = variable.value();
184  }
185 
186  // Launch the subprocess.
187  Option<Try<process::Subprocess>> external = None();
188 
189  if (command.shell()) {
190  // Use the shell variant.
191  if (!command.has_value()) {
192  promise.fail("Shell command is not specified");
193  return;
194  }
195 
196  VLOG(2) << "Launching health command '" << command.value() << "'";
197 
198  external = process::subprocess(
199  command.value(),
200  process::Subprocess::PATH("/dev/null"),
201  process::Subprocess::FD(STDERR_FILENO),
202  process::Subprocess::FD(STDERR_FILENO),
203  process::NO_SETSID,
204  environment);
205  } else {
206  // Use the exec variant.
207  if (!command.has_value()) {
208  promise.fail("Executable path is not specified");
209  return;
210  }
211 
212  vector<string> argv;
213  foreach (const string& arg, command.arguments()) {
214  argv.push_back(arg);
215  }
216 
217  VLOG(2) << "Launching health command [" << command.value() << ", "
218  << strings::join(", ", argv) << "]";
219 
220  external = process::subprocess(
221  command.value(),
222  argv,
223  process::Subprocess::PATH("/dev/null"),
224  process::Subprocess::FD(STDERR_FILENO),
225  process::Subprocess::FD(STDERR_FILENO),
226  process::NO_SETSID,
227  None(),
228  environment);
229  }
230 
231  CHECK_SOME(external);
232 
233  if (external.get().isError()) {
234  failure("Error creating subprocess for healthcheck: " +
235  external.get().error());
236  return;
237  }
238 
239  pid_t commandPid = external.get().get().pid();
240 
241  process::Future<Option<int>> status = external.get().get().status();
242  status.await(Seconds(check.timeout_seconds()));
243 
244  if (!status.isReady()) {
245  string msg = "Command check failed with reason: ";
246  if (status.isFailed()) {
247  msg += "failed with error: " + status.failure();
248  } else if (status.isDiscarded()) {
249  msg += "status future discarded";
250  } else {
251  msg += "status still pending after timeout " +
252  stringify(Seconds(check.timeout_seconds()));
253  }
254 
255  if (commandPid != -1) {
256  // Cleanup the external command process.
257  os::killtree(commandPid, SIGKILL);
258  VLOG(1) << "Kill health check command " << commandPid;
259  }
260 
261  failure(msg);
262  return;
263  }
264 
265  int statusCode = status.get().get();
266  if (statusCode != 0) {
267  string message = "Health command check " + WSTRINGIFY(statusCode);
268  failure(message);
269  } else {
270  success();
271  }
272  }
273 
274  void reschedule()
275  {
276  VLOG(1) << "Rescheduling health check in "
277  << Seconds(check.interval_seconds());
278 
279  delay(Seconds(check.interval_seconds()), self(), &Self::_healthCheck);
280  }

288  process::Time startTime;
289 };
