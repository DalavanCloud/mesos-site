41 // Resources come in three types: scalar, ranges, and sets. These are
42 // represented using protocol buffers. To make manipulation of
43 // resources easier within the Mesos core and for scheduler writers,
44 // we provide generic overloaded operators (see below) as well as a
45 // general Resources class that encapsulates a collection of protocol
46 // buffer Resource objects. The Resources class also provides a few
47 // static routines to allow parsing resources (e.g., from the command
48 // line), as well as determining whether or not a Resource object is
49 // valid. Note that many of these operations have not been optimized
50 // but instead just written for correct semantics.

166  // Returns the summed up Resources given a hashmap<Key, Resources>.
167  //
168  // NOTE: While scalar resources such as "cpus" sum correctly,
169  // non-scalar resources such as "ports" do not.
170  // e.g. "cpus:2" + "cpus:1" = "cpus:3"
171  // "ports:[0-100]" + "ports:[0-100]" = "ports:[0-100]"
172  //
173  // TODO(mpark): Deprecate this function once we introduce the
174  // concept of "cluster-wide" resources which provides correct
175  // semantics for summation over all types of resources. (e.g.
176  // non-scalar)
177  template <typename Key>
178  static Resources sum(const hashmap<Key, Resources>& _resources)
179  {
180  Resources result;
181 
182  foreachvalue (const Resources& resources, _resources) {
183  result += resources;
184  }
185 
186  return result;
187  }

264  // Certain offer operations (e.g., RESERVE, UNRESERVE, CREATE or
265  // DESTROY) alter the offered resources. The following methods

271  template <typename Iterable>
272  Try<Resources> apply(const Iterable& operations) const
273  {
274  Resources result = *this;
275 
276  foreach (const Offer::Operation& operation, operations) {
277  Try<Resources> transformed = result.apply(operation);
278  if (transformed.isError()) {
279  return Error(transformed.error());
280  }
281 
282  result = transformed.get();
283  }
284 
285  return result;
286  }
