57 Try<std::string> shell(const std::string& fmt, const T&... t)
58 {
59  const Try<std::string> command = strings::internal::format(fmt, t...);
60  if (command.isError()) {
61  return Error(command.error());
62  }
63 
64  FILE* file;
65  std::ostringstream stdout;
66 
67  if ((file = popen(command.get().c_str(), "r")) == NULL) {
68  return Error("Failed to run '" + command.get() + "'");
69  }
70 
71  char line[1024];
72  // NOTE(vinod): Ideally the if and while loops should be interchanged. But
73  // we get a broken pipe error if we don't read the output and simply close.
74  while (fgets(line, sizeof(line), file) != NULL) {
75  stdout << line;
76  }
77 
78  if (ferror(file) != 0) {
79  pclose(file); // Ignoring result since we already have an error.
80  return Error("Error reading output of '" + command.get() + "'");
81  }
82 
83  int status;
84  if ((status = pclose(file)) == -1) {
85  return Error("Failed to get status of '" + command.get() + "'");
86  }
87 
88  if (WIFSIGNALED(status)) {
89  return Error(
90  "Running '" + command.get() + "' was interrupted by signal '" +
91  strsignal(WTERMSIG(status)) + "'");
92  } else if ((WEXITSTATUS(status) != EXIT_SUCCESS)) {
93  LOG(ERROR) << "Command '" << command.get()
94  << "' failed; this is the output:\n" << stdout.str();
95  return Error(
96  "Failed to execute '" + command.get() + "'; the command was either "
97  "not found or exited with a non-zero exit status: " +
98  stringify(WEXITSTATUS(status)));
99  }
100 
101  return stdout.str();
102 }

117 // Executes a command by calling "/bin/sh -c <command>", and returns
118 // after the command has been completed. Returns 0 if succeeds, and
119 // return -1 on error (e.g., fork/exec/waitpid failed). This function
120 // is async signal safe. We return int instead of returning a Try
121 // because Try involves 'new', which is not async signal safe.
122 inline int system(const std::string& command)
123 {
124  pid_t pid = ::fork();
125 
126  if (pid == -1) {
127  return -1;
128  } else if (pid == 0) {
129  // In child process.
130  ::execlp(
131  Shell::name, Shell::arg0, Shell::arg1, command.c_str(), (char*)NULL);
132  ::exit(127);
133  } else {
134  // In parent process.
135  int status;
136  while (::waitpid(pid, &status, 0) == -1) {
137  if (errno != EINTR) {
138  return -1;
139  }
140  }
141 
142  return status;
143  }
144 }
