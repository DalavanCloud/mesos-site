21 // Constructs a Markdown based help "page" for a route with the
22 // following template:
23 //
24 // ### TL;DR; ###
25 // tldr
26 //
27 // ### USAGE ###
28 // usage
29 //
30 // ### DESCRIPTION ###
31 // description
32 //
33 // references
34 //
35 // See the 'TLDR', 'USAGE', 'DESCRIPTION', and 'REFERENCES' helpers
36 // below to more easily construct your help pages.
37 inline std::string HELP(
38  std::string tldr,
39  std::string usage,
40  std::string description,
41  const Option<std::string>& references = None())
42 {
43  // Make sure 'tldr', 'usage', and 'description' end with a newline.
44  if (!strings::endsWith(tldr, "\n")) {
45  tldr += "\n";
46  }
47 
48  if (!strings::endsWith(usage, "\n")) {
49  usage += "\n";
50  }
51 
52  if (!strings::endsWith(description, "\n")) {
53  description += "\n";
54  }
55 
56  // Construct the help string.
57  std::string help =
58  "### TL;DR; ###\n" +
59  tldr +
60  "\n" +
61  "### USAGE ###\n" +
62  usage +
63  "\n" +
64  "### DESCRIPTION ###\n" +
65  description;
66 
67  if (references.isSome()) {
68  help += "\n";
69  help += references.get();
70  }
71 
72  return help;
73 }
