29 // These two macros create a Factory class that constructs instances of
30 // subclasses of a type T, whose constructors take a parameter of type P,
31 // based on a registered name for each subclass. To use them:
32 //
33 // 1) In a header file, call DECLARE_FACTORY(T, P).
34 //
35 // 2) In a source file, call DEFINE_FACTORY(T, P) { block }, where the block
36 // calls registerClass<C>("name") for each subclass C of T to register.
37 //
38 // 3) You can now call TFactory::instantiate("name", p) to create an instance
39 // of the class registered with a given name (or NULL if none exists).
40 //
41 // Note: Having to register all the classes in one file isn't ideal, but it
42 // seems to be the most foolproof solution. If we also want to allow classes
43 // to be loaded through shared libraries, it would be possible to add an init
44 // function to the library that registers them and make the registerClass()
45 // Another method people use to register classes is to have a static object
46 // whose constructor adds the class to the factory, but this method doesn't
47 // work without special care when the class is in a static library since the
48 // unreferenced static object won't be included by the compiler.

96 template<typename T, typename P> class Factory {
97  std::map<std::string, Creator<T, P> *> creators;
98 
99 protected:
100  template<typename C> void registerClass(const std::string& name) {
101  if (creators.find(name) != creators.end())
102  fatal("Two classes registered with name \"%s\"", name.c_str());
103  creators[name] = new ConcreteCreator<C, T, P>();
104  }

106  // Note: This instantiate needs to have a different name than the

108  T * instantiate2(const std::string& name, P p) {
109  if (creators.find(name) == creators.end())
110  return NULL;
111  else
