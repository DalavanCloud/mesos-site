57 // Returns true if the future becomes ready, discarded, or failed
58 // after the wait.
59 template <typename T>
60 bool await(const process::Future<T>& future, const Duration& duration)
61 {
62  if (!process::Clock::paused()) {
63  return future.await(duration);
64  }
65 
66  // If the clock is paused, no new timers will expire.
67  // Future::await(duration) may hang forever because it depends on
68  // a timer to expire after 'duration'. We instead ensure all
69  // expired timers are flushed and check if the future is satisfied.
70  Stopwatch stopwatch;
71  stopwatch.start();
72 
73  // Settle to make sure all expired timers are executed (not
74  // necessarily finished, see below).
75  process::Clock::settle();
76 
77  while (future.isPending() && stopwatch.elapsed() < duration) {
78  // Synchronous operations and asynchronous process::Process
79  // operations should finish when the above 'settle()' returns.
80  // Other types of async operations such as io::write() may not.
81  // Therefore we wait the specified duration for it to complete.
82  // Note that nothing prevents the operations to schedule more
83  // timeouts for some time in the future. These timeouts will
84  // never be executed due to the paused process::Clock. In this
85  // case we return after the stopwatch (system clock) runs out.
86  os::sleep(Milliseconds(10));
87  }
88 
89  return !future.isPending();
90 }

112 template <typename T>
113 ::testing::AssertionResult AwaitAssertReady(
114  const char* expr,
115  const char*, // Unused string representation of 'duration'.
116  const process::Future<T>& actual,
117  const Duration& duration)
118 {
119  if (!process::internal::await(actual, duration)) {
120  return ::testing::AssertionFailure()
121  << "Failed to wait " << duration << " for " << expr;
122  } else if (actual.isDiscarded()) {
123  return ::testing::AssertionFailure()
124  << expr << " was discarded";
125  } else if (actual.isFailed()) {
126  return ::testing::AssertionFailure()
127  << "(" << expr << ").failure(): " << actual.failure();
128  }
129 
130  return ::testing::AssertionSuccess();
131 }
