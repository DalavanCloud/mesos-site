104  inline std::string basename() const
105  {
106  if (value.empty()) {
107  return std::string(".");
108  }
109 
110  size_t end = value.size() - 1;
111 
112  // Remove trailing slashes.
113  if (value[end] == '/') {
114  end = value.find_last_not_of('/', end);
115 
116  // Paths containing only slashes result into "/".
117  if (end == std::string::npos) {
118  return std::string("/");
119  }
120  }
121 
122  // 'start' should point towards the character after the last slash
123  // that is non trailing.

167  if (value[end] == '/') {

174  // Paths containing no slashes result in ".".
175  if (end == std::string::npos) {
176  return std::string(".");

180  if (end == 0) {
