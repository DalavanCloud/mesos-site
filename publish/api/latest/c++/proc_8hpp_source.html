257 inline Result<std::string> cmdline(const Option<pid_t>& pid = None())
258 {
259  const std::string path = pid.isSome()
260  ? "/proc/" + stringify(pid.get()) + "/cmdline"
261  : "/proc/cmdline";
262 
263  std::ifstream file(path.c_str());
264 
265  if (!file.is_open()) {
266  // Need to check if file exists AFTER we open it to guarantee
267  // process hasn't terminated (or if it has, we at least have a
268  // file which the kernel _should_ respect until a close).
269  if (!os::exists(path)) {
270  return None();
271  }
272  return Error("Failed to open '" + path + "'");
273  }
274 
275  std::stringbuf buffer;
276 
277  do {
278  // Read each argument in "argv", separated by null bytes.
279  file.get(buffer, '\0');
280 
281  // Check for any read errors.
282  if (file.fail() && !file.eof()) {
283  file.close();
284  return Error("Failed to read '" + path + "'");
285  } else if (!file.eof()) {
286  file.get(); // Read the null byte.
287  buffer.sputc(' '); // Put a space between each command line argument.
288  }
289  } while (!file.eof());
290 
291  return buffer.str();
292 }

440 // Reads from /proc/cpuinfo and returns a list of CPUs.
441 inline Try<std::list<CPU> > cpus()
442 {
443  std::list<CPU> results;
444 
445  std::ifstream file("/proc/cpuinfo");
446 
447  if (!file.is_open()) {
448  return Error("Failed to open /proc/cpuinfo");
449  }
450 
451  // Placeholders as we parse the file.
452  Option<unsigned int> id;
453  Option<unsigned int> core;
454  Option<unsigned int> socket;
455 
456  std::string line;
457  while (std::getline(file, line)) {
458  if (line.find("processor") == 0 ||
459  line.find("physical id") == 0 ||
460  line.find("core id") == 0) {
461  // Get out and parse the value.
462  std::vector<std::string> tokens = strings::tokenize(line, ": ");
463 
464  if (tokens.size() < 2) {
465  return Error("Unexpected format in /proc/cpuinfo: " +
466  stringify(tokens));
467  }
468 
469  Try<unsigned int> value = numify<unsigned int>(tokens.back());
470  if (value.isError()) {
471  return Error(value.error());
472  }
473 
474  // Now save the value.
475  if (line.find("processor") == 0) {
476  if (id.isSome()) {
477  // The physical id and core id are not present in this case.
478  results.push_back(CPU(id.get(), 0, 0));
479  }
480  id = value.get();
481  } else if (line.find("physical id") == 0) {
482  if (socket.isSome()) {
483  return Error("Unexpected format in /proc/cpuinfo");
484  }
485  socket = value.get();
486  } else if (line.find("core id") == 0) {
487  if (core.isSome()) {
488  return Error("Unexpected format in /proc/cpuinfo");
489  }
490  core = value.get();
491  }
492 
493  // And finally create a CPU if we have all the information.
494  if (id.isSome() && core.isSome() && socket.isSome()) {
495  results.push_back(CPU(id.get(), core.get(), socket.get()));
496  id = None();
497  core = None();
498  socket = None();
499  }
500  }
501  }
502 
503  // Add the last processor if the physical id and core id were not present.
504  if (id.isSome()) {
505  // The physical id and core id are not present.
506  results.push_back(CPU(id.get(), 0, 0));
507  }
508 
509  if (file.fail() && !file.eof()) {
510  file.close();
511  return Error("Failed to read /proc/cpuinfo");
512  }
513 
514  file.close();
515 
516  return results;
517 }
