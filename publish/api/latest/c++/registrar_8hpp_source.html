39 // Defines an abstraction for operations that can be applied on the
40 // Registry.
41 // TODO(xujyan): Make Operation generic so that we can apply them
42 // against a generic "batch operation applier" abstraction, see TODO
43 // below for more details.
44 class Operation : public process::Promise<bool>
45 {
46 public:
47  Operation() : success(false) {}
48  virtual ~Operation() {}
49 
50  // Attempts to invoke the operation on the registry object.
51  // Aided by accumulator(s):
52  // slaveIDs - is the set of registered slaves.
53  //
54  // NOTE: the "strict" parameter only applies to operations that
55  // affect slaves (i.e. registration). See Flags::registry_strict
56  // in master/flags.cpp for more information.
57  //
58  // Returns whether the operation mutates 'registry', or an error if
59  // the operation cannot be applied successfully.
60  Try<bool> operator()(
61  Registry* registry,
62  hashset<SlaveID>* slaveIDs,
63  bool strict)
64  {
65  const Try<bool> result = perform(registry, slaveIDs, strict);
66 
67  success = !result.isError();
68 
69  return result;
70  }
71 
72  // Sets the promise based on whether the operation was successful.
73  bool set() { return process::Promise<bool>::set(success); }
74 
75 protected:
76  virtual Try<bool> perform(
77  Registry* registry,
78  hashset<SlaveID>* slaveIDs,
79  bool strict) = 0;
80 
81 private:
82  bool success;
83 };

86 // TODO(xujyan): Add a generic abstraction for applying batches of
87 // operations against State Variables. The Registrar and other
88 // components could leverage this. This abstraction would be
89 // templatized to take the type, along with any accumulators:
90 // template <typename T,
91 // typename X = None,
92 // typename Y = None,
93 // typename Z = None>
94 // T: the data type that the batch operations can be applied on.
95 // X, Y, Z: zero to 3 generic accumulators that facilitate the batch
96 // of operations.
97 // This allows us to reuse the idea of "doing batches of operations"
98 // on other potential new state variables (i.e. repair state, offer
99 // reservations, etc).
100 class Registrar
101 {
102 public:
103  // If flags.registry_strict is true, all operations will be
104  // permitted.
105  Registrar(const Flags& flags, state::protobuf::State* state);
106  ~Registrar();
107 
108  // Recovers the Registry, persisting the new Master information.
109  // The Registrar must be recovered to allow other operations to
110  // proceed.
111  // TODO(bmahler): Consider a "factory" for constructing the
112  // Registrar, to eliminate the need for passing 'MasterInfo'.
113  // This is required as the Registrar is injected into the Master,
114  // and therefore MasterInfo is unknown during construction.
115  process::Future<Registry> recover(const MasterInfo& info);

117  // Applies an operation on the Registry.

123  process::Future<bool> apply(process::Owned<Operation> operation);

125 private:
126  RegistrarProcess* process;
