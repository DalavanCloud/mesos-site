503 // By default, recursively deletes a directory akin to: 'rm -r'. If the
504 // programmer sets recursive to false, it deletes a directory akin to: 'rmdir'.
505 // Note that this function expects an absolute path.
506 inline Try<Nothing> rmdir(const std::string& directory, bool recursive = true)
507 {
508  if (!recursive) {
509  if (::rmdir(directory.c_str()) < 0) {
510  return ErrnoError();
511  }
512  } else {
513  char* paths[] = {const_cast<char*>(directory.c_str()), NULL};
514 
515  FTS* tree = fts_open(paths, FTS_NOCHDIR, NULL);
516  if (tree == NULL) {
517  return ErrnoError();
518  }
519 
520  FTSENT* node;
521  while ((node = fts_read(tree)) != NULL) {
522  switch (node->fts_info) {
523  case FTS_DP:
524  if (::rmdir(node->fts_path) < 0 && errno != ENOENT) {
525  return ErrnoError();
526  }
527  break;
528  case FTS_F:
529  case FTS_SL:
530  if (::unlink(node->fts_path) < 0 && errno != ENOENT) {
531  return ErrnoError();
532  }
533  break;
534  default:
535  break;
536  }
537  }
538 
539  if (errno != 0) {
540  return ErrnoError();
541  }
542 
543  if (fts_close(tree) < 0) {
544  return ErrnoError();
545  }
546  }
547 
548  return Nothing();
549 }

552 // Executes a command by calling "/bin/sh -c <command>", and returns
553 // after the command has been completed. Returns 0 if succeeds, and
554 // return -1 on error (e.g., fork/exec/waitpid failed). This function
555 // is async signal safe. We return int instead of returning a Try
556 // because Try involves 'new', which is not async signal safe.
557 inline int system(const std::string& command)
558 {
559  pid_t pid = ::fork();
560 
561  if (pid == -1) {
562  return -1;
563  } else if (pid == 0) {
564  // In child process.
565  ::execl("/bin/sh", "sh", "-c", command.c_str(), (char*) NULL);
566  ::exit(127);
567  } else {
568  // In parent process.
569  int status;
570  while (::waitpid(pid, &status, 0) == -1) {
571  if (errno != EINTR) {
572  return -1;
573  }
574  }
575 
576  return status;
577  }
578 }

635 inline Result<uid_t> getuid(const Option<std::string>& user = None())
636 {
637  if (user.isNone()) {
638  return ::getuid();
639  }
640 
641  struct passwd passwd;
642  struct passwd* result = NULL;
643 
644  int size = sysconf(_SC_GETPW_R_SIZE_MAX);
645  if (size == -1) {
646  // Initial value for buffer size.
647  size = 1024;
648  }
649 
650  while (true) {
651  char* buffer = new char[size];
652 
653  if (getpwnam_r(user.get().c_str(), &passwd, buffer, size, &result) == 0) {
654  // The usual interpretation of POSIX is that getpwnam_r will
655  // return 0 but set result == NULL if the user is not found.
656  if (result == NULL) {
657  delete[] buffer;
658  return None();
659  }
660 
661  uid_t uid = passwd.pw_uid;
662  delete[] buffer;
663  return uid;
664  } else {
665  // RHEL7 (and possibly other systems) will return non-zero and
666  // set one of the following errors for "The given name or uid
667  // was not found." See 'man getpwnam_r'. We only check for the
668  // errors explicitly listed, and do not consider the ellipsis.
669  if (errno == ENOENT ||
670  errno == ESRCH ||
671  errno == EBADF ||
672  errno == EPERM) {
673  return None();
674  }
675 
676  if (errno != ERANGE) {
677  delete[] buffer;
678  return ErrnoError("Failed to get username information");
679  }
680  // getpwnam_r set ERANGE so try again with a larger buffer.
681  size *= 2;
682  delete[] buffer;
683  }
684  }
685 
686  return Result<uid_t>(UNREACHABLE());
687 }
