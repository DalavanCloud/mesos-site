41 // An abstraction of "state" (possibly between multiple distributed
42 // components) represented by "variables" (effectively key/value
43 // pairs). Variables are versioned such that setting a variable in the
44 // state will only succeed if the variable has not changed since last
45 // fetched. Varying implementations of state provide varying
46 // replicated guarantees.
47 //
48 // Note that the semantics of 'fetch' and 'store' provide
49 // atomicity. That is, you can not store a variable that has changed
50 // since you did the last fetch. That is, if a store succeeds then no
51 // other writes have been performed on the variable since your fetch.
52 //
53 // Example:
54 //
55 // Storage* storage = new ZooKeeperStorage();
56 // State* state = new State(storage);
57 // Future<Variable> variable = state->fetch("slaves");
58 // std::string value = update(variable.value());
59 // variable = variable.mutate(value);
60 // state->store(variable);
61

155 inline process::Future<Option<Variable> > State::store(const Variable& variable)
156 {
157  // Note that we try and swap an entry even if the value didn't change!
158  UUID uuid = UUID::fromBytes(variable.entry.uuid());
159 
160  // Create a new entry to replace the existing entry provided the
161  // UUID matches.
162  Entry entry;
163  entry.set_name(variable.entry.name());
164  entry.set_uuid(UUID::random().toBytes());
165  entry.set_value(variable.entry.value());
166 
167  return storage->set(entry, uuid)
168  .then(lambda::bind(&State::_store, entry, lambda::_1));
169 }
