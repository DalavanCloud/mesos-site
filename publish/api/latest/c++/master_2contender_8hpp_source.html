51 // An abstraction for contending to be a leading master.
52 class MasterContender
53 {
54 public:
55  // Attempts to create a master contender using the specified
56  // Zookeeper.
57  // The Zookeeper address should be one of:
58  // - zk://host1:port1,host2:port2,.../path
59  // - zk://username:password@host1:port1,host2:port2,.../path
60  // - file:///path/to/file (where file contains one of the above)
61  // Note that the returned contender still needs to be 'initialize()'d.
62  static Try<MasterContender*> create(const std::string& zk);
63 
64  // Note that the contender's membership, if obtained, is scheduled
65  // to be cancelled during destruction.
66  virtual ~MasterContender() = 0;
67 
68  // Initializes the contender with the MasterInfo of the master it
69  // contends on behalf of.
70  virtual void initialize(const MasterInfo& masterInfo) = 0;
71 
72  // Returns a Future<Nothing> once the contender has entered the
73  // contest (by obtaining a membership) and an error otherwise.
74  // A failed future is returned if this method is called before
75  // initialize().
76  // The inner Future returns Nothing when the contender is out of
77  // the contest (i.e. its membership is lost).
78  //
79  // This method can be used to contend again after candidacy is
80  // obtained (the outer future satisfied), otherwise the future for
81  // the pending election is returned.
82  // Recontending after candidacy is obtained causes the previous
83  // candidacy to be withdrawn.
84  virtual process::Future<process::Future<Nothing> > contend() = 0;
85 };
