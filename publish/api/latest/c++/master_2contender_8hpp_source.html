49 // An abstraction for contending to be a leading master.
50 //
51 // TODO(benh): Support contending with a v1::MasterInfo.
52 class MasterContender
53 {
54 public:
55  // Attempts to create a master contender using the specified
56  // mechanism.
57  // The mechanism address should be one of:
58  // - zk://host1:port1,host2:port2,.../path
59  // - zk://username:password@host1:port1,host2:port2,.../path
60  // Note that the returned contender still needs to be 'initialize()'d.
61  static Try<MasterContender*> create(const Option<std::string>& _mechanism);
62 
63  // Note that the contender's membership, if obtained, is scheduled
64  // to be cancelled during destruction.
65  virtual ~MasterContender() = 0;
66 
67  // Initializes the contender with the MasterInfo of the master it
68  // contends on behalf of.
69  virtual void initialize(const MasterInfo& masterInfo) = 0;
70 
71  // Returns a Future<Nothing> once the contender has entered the
72  // contest (by obtaining a membership) and an error otherwise.
73  // A failed future is returned if this method is called before
74  // initialize().
75  // The inner Future returns Nothing when the contender is out of
76  // the contest (i.e. its membership is lost).
77  //
78  // This method can be used to contend again after candidacy is
79  // obtained (the outer future satisfied), otherwise the future for
80  // the pending election is returned.
81  // Recontending after candidacy is obtained causes the previous
82  // candidacy to be withdrawn.
83  virtual process::Future<process::Future<Nothing> > contend() = 0;
84 };
