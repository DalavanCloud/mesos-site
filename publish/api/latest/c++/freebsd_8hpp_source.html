29 inline Result<Process> process(pid_t pid)
30 {
31  // KERN_PROC_PID fails for zombies, so we fetch the whole process table and
32  // find our process manually.
33 
34  const Try<std::vector<kinfo_proc>> kinfos =
35  os::sysctl(CTL_KERN, KERN_PROC, KERN_PROC_ALL).table();
36 
37  if (kinfos.isError()) {
38  return Error("Failed to retrieve process table via sysctl: " +
39  kinfos.error());
40  }
41 
42  foreach (const kinfo_proc& kinfo, kinfos.get()) {
43  if (kinfo.ki_pid == pid) {
44  int pagesize = getpagesize();
45  return Process(kinfo.ki_pid,
46  kinfo.ki_ppid,
47  kinfo.ki_pgid,
48  kinfo.ki_sid,
49  kinfo.ki_rssize * pagesize,
50  kinfo.ki_rusage.ru_utime,
51  kinfo.ki_rusage.ru_stime,
52  kinfo.ki_comm,
53  kinfo.ki_stat == SZOMB);
54  }
55  }
56 
57  return None();
58 }
