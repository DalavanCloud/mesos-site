33  if (recursive) {
34  // TODO(bmahler): Consider walking the file tree instead. We would need

36  std::string command =
37  "chown -R " + stringify(uid) + ':' + stringify(gid) + " '" + path + "'";
38 
39  int status = os::system(command);
40  if (status != 0) {
41  return ErrnoError(

44  }
45  } else {

47  return ErrnoError();

55 // Changes the specified path's user and group ownership to that of
56 // the specified user.
57 inline Try<Nothing> chown(
58  const std::string& user,
59  const std::string& path,
60  bool recursive = true)
61 {
62  passwd* passwd;
63  if ((passwd = ::getpwnam(user.c_str())) == NULL) {
64  return ErrnoError("Failed to get user information for '" + user + "'");
65  }
66 
67  return chown(passwd->pw_uid, passwd->pw_gid, path, recursive);
68 }
