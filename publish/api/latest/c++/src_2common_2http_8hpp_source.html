58 // Deserializes a string message into a protobuf message based on the
59 // HTTP content type.
60 template <typename Message>
61 Try<Message> deserialize(
62  ContentType contentType,
63  const std::string& body)
64 {
65  switch (contentType) {
66  case ContentType::PROTOBUF: {
67  Message message;
68  if (!message.ParseFromString(body)) {
69  return Error("Failed to parse body into a protobuf object");
70  }
71  return message;
72  }
73  case ContentType::JSON: {
74  Try<JSON::Value> value = JSON::parse(body);
75  if (value.isError()) {
76  return Error("Failed to parse body into JSON: " + value.error());
77  }
78 
79  return ::protobuf::parse<Message>(value.get());
80  }
81  }
