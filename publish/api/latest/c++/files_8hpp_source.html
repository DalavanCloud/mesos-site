70 // Returns our JSON representation of a file or directory.
71 // The JSON contains all of the information one would find in ls -l.
72 // Example JSON:
73 // {
74 // 'path': '\/some\/file',
75 // 'mode': '-rwxrwxrwx',
76 // 'nlink': 5,
77 // 'uid': 'bmahler',
78 // 'gid': 'employee',
79 // 'size': 4096, // Bytes.
80 // 'mtime': 1348258116, // Unix timestamp.
81 // }
82 inline JSON::Object jsonFileInfo(const std::string& path,
83  const struct stat& s)
84 {
85  JSON::Object file;
86  file.values["path"] = path;
87  file.values["nlink"] = s.st_nlink;
88  file.values["size"] = s.st_size;
89  file.values["mtime"] = s.st_mtime;
90 
91  char filetype;
92  if (S_ISREG(s.st_mode)) {
93  filetype = '-';
94  } else if (S_ISDIR(s.st_mode)) {
95  filetype = 'd';
96  } else if (S_ISCHR(s.st_mode)) {
97  filetype = 'c';
98  } else if (S_ISBLK(s.st_mode)) {
99  filetype = 'b';
100  } else if (S_ISFIFO(s.st_mode)) {
101  filetype = 'p';
102  } else if (S_ISLNK(s.st_mode)) {
103  filetype = 'l';
104  } else if (S_ISSOCK(s.st_mode)) {
105  filetype = 's';
106  } else {
107  filetype = '-';
108  }
109 
110  int owner = (s.st_mode & 0700) >> 6;
111  int group = (s.st_mode & 0070) >> 3;
112  int other = s.st_mode & 0007;
113 
114  file.values["mode"] = strings::format("%c%c%c%c%c%c%c%c%c%c",
115  filetype,
116  (owner & 0x4) ? 'r' : '-',
117  (owner & 0x2) ? 'w' : '-',
118  (owner & 0x1) ? 'x' : '-',
119  (group & 0x4) ? 'r' : '-',
120  (group & 0x2) ? 'w' : '-',
121  (group & 0x1) ? 'x' : '-',
122  (other & 0x4) ? 'r' : '-',
123  (other & 0x2) ? 'w' : '-',
124  (other & 0x1) ? 'x' : '-').get();
125 
126  passwd* p = getpwuid(s.st_uid);
127  if (p != NULL) {
128  file.values["uid"] = p->pw_name;
129  } else {
130  file.values["uid"] = stringify(s.st_uid);
131  }
132 
133  struct group* g = getgrgid(s.st_gid);
134  if (g != NULL) {
135  file.values["gid"] = g->gr_name;
136  } else {
137  file.values["gid"] = stringify(s.st_gid);
138  }
139 
140  return file;
141 }
