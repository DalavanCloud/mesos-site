170 inline Result<uid_t> getuid(const Option<std::string>& user = None())
171 {
172  if (user.isNone()) {
173  return ::getuid();
174  }
175 
176  struct passwd passwd;
177  struct passwd* result = NULL;
178 
179  int size = sysconf(_SC_GETPW_R_SIZE_MAX);
180  if (size == -1) {
181  // Initial value for buffer size.
182  size = 1024;
183  }
184 
185  while (true) {
186  char* buffer = new char[size];
187 
188  if (getpwnam_r(user.get().c_str(), &passwd, buffer, size, &result) == 0) {
189  // The usual interpretation of POSIX is that getpwnam_r will
190  // return 0 but set result == NULL if the user is not found.
191  if (result == NULL) {
192  delete[] buffer;
193  return None();
194  }
195 
196  uid_t uid = passwd.pw_uid;
197  delete[] buffer;
198  return uid;
199  } else {
200  // RHEL7 (and possibly other systems) will return non-zero and
201  // set one of the following errors for "The given name or uid
202  // was not found." See 'man getpwnam_r'. We only check for the
203  // errors explicitly listed, and do not consider the ellipsis.
204  if (errno == ENOENT ||
205  errno == ESRCH ||
206  errno == EBADF ||
207  errno == EPERM) {
208  delete[] buffer;
209  return None();
210  }
211 
212  if (errno != ERANGE) {
213  delete[] buffer;
214  return ErrnoError("Failed to get username information");
215  }
216  // getpwnam_r set ERANGE so try again with a larger buffer.
217  size *= 2;
218  delete[] buffer;
219  }
220  }
221 
222  UNREACHABLE();
223 }

282 inline Try<Nothing> su(const std::string& user)
283 {
284  Result<gid_t> gid = os::getgid(user);
285  if (gid.isError() || gid.isNone()) {
286  return Error("Failed to getgid: " +
287  (gid.isError() ? gid.error() : "unknown user"));
288  } else if (::setgid(gid.get())) {
289  return ErrnoError("Failed to set gid");
290  }
291 
292  // Set the supplementary group list. We ignore EPERM because
293  // performing a no-op call (switching to same group) still
294  // requires being privileged, unlike 'setgid' and 'setuid'.
295  if (::initgroups(user.c_str(), gid.get()) == -1 && errno != EPERM) {
296  return ErrnoError("Failed to set supplementary groups");
297  }
298 
299  Result<uid_t> uid = os::getuid(user);
300  if (uid.isError() || uid.isNone()) {
301  return Error("Failed to getuid: " +
302  (uid.isError() ? uid.error() : "unknown user"));
303  } else if (::setuid(uid.get())) {
304  return ErrnoError("Failed to setuid");
305  }
306 
307  return Nothing();

354 // Suspends execution for the given duration.
355 inline Try<Nothing> sleep(const Duration& duration)
356 {
357  timespec remaining;
358  remaining.tv_sec = static_cast<long>(duration.secs());
359  remaining.tv_nsec =
360  static_cast<long>((duration - Seconds(remaining.tv_sec)).ns());
361 
362  while (nanosleep(&remaining, &remaining) == -1) {
363  if (errno == EINTR) {
364  continue;
365  } else {
366  return ErrnoError();
367  }
368  }
369 
370  return Nothing();
371 }

432 // Returns the total size of main and free memory.
433 inline Try<Memory> memory()
434 {
435  Memory memory;
436 
437 // TODO(dforsyth): Refactor these implementations into seperate, platform
438 // specific files.
439 #ifdef __linux__
440  struct sysinfo info;
441  if (sysinfo(&info) != 0) {
442  return ErrnoError();
443  }
444 
445 # if LINUX_VERSION_CODE >= KERNEL_VERSION(2, 3, 23)
446  memory.total = Bytes(info.totalram * info.mem_unit);
447  memory.free = Bytes(info.freeram * info.mem_unit);
448  memory.totalSwap = Bytes(info.totalswap * info.mem_unit);
449  memory.freeSwap = Bytes(info.freeswap * info.mem_unit);
450 # else
451  memory.total = Bytes(info.totalram);
452  memory.free = Bytes(info.freeram);
453  memory.totalSwap = Bytes(info.totalswap);
454  memory.freeSwap = Bytes(info.freeswap);
455 # endif
456 
457  return memory;
458 
459 #elif defined __APPLE__
460  const Try<int64_t> totalMemory = os::sysctl(CTL_HW, HW_MEMSIZE).integer();
461 
462  if (totalMemory.isError()) {
463  return Error(totalMemory.error());
464  }
465  memory.total = Bytes(totalMemory.get());
466 
467  // Size of free memory is available in terms of number of
468  // free pages on Mac OS X.
469  const long pageSize = sysconf(_SC_PAGESIZE);
470  if (pageSize < 0) {
471  return ErrnoError();
472  }
473 
474  unsigned int freeCount;
475  size_t length = sizeof(freeCount);
476 
477  if (sysctlbyname(
478  "vm.page_free_count",
479  &freeCount,
480  &length,
481  NULL,
482  0) != 0) {
483  return ErrnoError();
484  }
485  memory.free = Bytes(freeCount * pageSize);
486 
487  struct xsw_usage usage;
488  length = sizeof(struct xsw_usage);
489  if (sysctlbyname(
490  "vm.swapusage",
491  &usage,
492  &length,
493  NULL,
494  0) != 0) {
495  return ErrnoError();
496  }
497  memory.totalSwap = Bytes(usage.xsu_total * pageSize);
498  memory.freeSwap = Bytes(usage.xsu_avail * pageSize);
499 
500  return memory;
501 
502 #elif defined __FreeBSD__
503  const Try<int64_t> physicalMemory = os::sysctl(CTL_HW, HW_PHYSMEM).integer();
504  if (physicalMemory.isError()) {
505  return Error(physicalMemory.error());
506  }
507  memory.total = Bytes(physicalMemory.get());
508 
509  const int pageSize = getpagesize();
510 
511  unsigned int freeCount;
512  size_t length = sizeof(freeCount);
513 
514  if (sysctlbyname(
515  "vm.stats.v_free_count",
516  &freeCount,
517  &length,
518  NULL,
519  0) != 0) {
520  return ErrnoError();
521  }
522  memory.free = Bytes(freeCount * pageSize);
523 
524  int totalBlocks = 0;
525  int usedBlocks = 0;
526 
527  int mib[3];
528  size_t mibSize = 2;
529  if (::sysctlnametomib("vm.swap_info", mib, &mibSize) != 0) {
530  return ErrnoError();
531  }
532 
533  // FreeBSD supports multiple swap devices. Here we sum across all of them.
534  struct xswdev xswd;
535  size_t xswdSize = sizeof(xswd);
536  int* mibDevice = &(mib[mibSize + 1]);
537  for (*mibDevice = 0; ; (*mibDevice)++) {
538  if (::sysctl(mib, 3, &xswd, &xswdSize, NULL, 0) != 0) {
539  if (errno == ENOENT) {
540  break;
541  }
542  return ErrnoError();
543  }
544 
545  totalBlocks += xswd.xsw_nblks;
546  usedBlocks += xswd.xsw_used;
547  }
548 
549  memory.totalSwap = Bytes(totalBlocks * pageSize);
550  memory.freeSwap = Bytes((totalBlocks - usedBlocks) * pageSize);
551 
552  return memory;
553 #else
554  return Error("Cannot determine the size of total and free memory");
555 #endif
556 }

599 // Overload of os::pids for filtering by groups and sessions.
600 // A group / session id of 0 will fitler on the group / session ID
601 // of the calling process.
602 inline Try<std::set<pid_t>> pids(Option<pid_t> group, Option<pid_t> session)
603 {
604  if (group.isNone() && session.isNone()) {
605  return os::pids();
606  } else if (group.isSome() && group.get() < 0) {
607  return Error("Invalid group");
608  } else if (session.isSome() && session.get() < 0) {
609  return Error("Invalid session");
610  }
611 
612  const Try<std::list<Process>> processes = os::processes();
613 
614  if (processes.isError()) {
615  return Error(processes.error());
616  }
617 
618  // Obtain the calling process group / session ID when 0 is provided.
619  if (group.isSome() && group.get() == 0) {
620  group = getpgid(0);
621  }
622  if (session.isSome() && session.get() == 0) {
623  session = getsid(0);
624  }
625 
626  std::set<pid_t> result;
627  foreach (const Process& process, processes.get()) {
628  // Group AND Session (intersection).
629  if (group.isSome() && session.isSome()) {
630  if (group.get() == process.group &&
631  process.session.isSome() &&
632  session.get() == process.session.get()) {
633  result.insert(process.pid);
634  }
635  } else if (group.isSome() && group.get() == process.group) {
636  result.insert(process.pid);
637  } else if (session.isSome() && process.session.isSome() &&
638  session.get() == process.session.get()) {
639  result.insert(process.pid);
640  }
641  }
642 
643  return result;
644 }

746 inline std::set<pid_t> children(
747  pid_t pid,
748  const std::list<Process>& processes,
749  bool recursive = true)
750 {
751  // Perform a breadth first search for descendants.
752  std::set<pid_t> descendants;
753  std::queue<pid_t> parents;
754  parents.push(pid);
755 
756  do {
757  pid_t parent = parents.front();
758  parents.pop();
759 
760  // Search for children of parent.
761  foreach (const Process& process, processes) {
762  if (process.parent == parent) {
763  // Have we seen this child yet?
