39 inline Result<Process> process(pid_t pid)
40 {
41  const Try<std::vector<kinfo_proc> >& processes =
42  os::sysctl(CTL_KERN, KERN_PROC, KERN_PROC_PID, pid).table(1);
43 
44  if (processes.isError()) {
45  return Error("Failed to get process via sysctl: " + processes.error());
46  } else if (processes.get().size() != 1) {
47  return None();
48  }
49 
50  const kinfo_proc process = processes.get()[0];
51 
52  // The command line from 'process.kp_proc.p_comm' only includes the
53  // first 16 characters from "arg0" (i.e., the canonical executable
54  // name). We can try to get "argv" via some sysctl magic. This first
55  // requires determining "argc" via KERN_PROCARGS2 followed by the
56  // actual arguments via KERN_PROCARGS. This is still insufficient
57  // with insufficient privilege (e.g., not being root). If we were
58  // only interested in the "executable path" (i.e., the first
59  // argument to 'exec' but none of the arguments) we could use
60  // proc_pidpath() instead.
61  Option<std::string> command = None();
62 
63  // Alternative implemenations using KERN_PROCARGS2:
64  // https://bitbucket.org/flub/psi-win32/src/d38f288de3b8/src/arch/macosx/macosx_process.c#cl-552
65  // https://gist.github.com/nonowarn/770696
66 
67 #ifdef KERN_PROCARGS2
68  // Looking at the source code of XNU (the Darwin kernel for OS X:
69  // www.opensource.apple.com/source/xnu/xnu-1699.24.23/bsd/kern/kern_sysctl.c),
70  // it appears as though KERN_PROCARGS2 writes 'argc' as the first
71  // word of the returned bytes.
72  Try<std::string> args = os::sysctl(CTL_KERN, KERN_PROCARGS2, pid).string();
73 
74  if (args.isSome()) {
75  int argc = *((int*) args.get().data());
76 
77  if (argc > 0) {
78  // Now grab the arguments.
79  args = os::sysctl(CTL_KERN, KERN_PROCARGS, pid).string();
80 
81  if (args.isSome()) {
82  // At this point 'args' contains the parameters to 'exec'
83  // delimited by null bytes, i.e., "executable path", then
84  // "arg0" (the canonical executable name), then "arg1", then
85  // "arg2", etc. Sometimes there are no arguments (argc = 1) so
86  // all we care about is the "executable path", but when there
87  // are arguments we grab "arg0" and on assuming that "arg0"
88  // really is the canonical executable name.
89 
90  // Tokenize the args by the null byte ('\0').
91  std::vector<std::string> tokens =
92  strings::tokenize(args.get(), std::string(1, '\0'));
93 
94  if (!tokens.empty()) {
95  if (argc == 1) {
96  // When there are no arguments, all we care about is the
97  // "executable path".
98  command = tokens[0];
99  } else if (argc > 1) {
100  // When there are arguments, we skip the "executable path"
101  // and just grab "arg0" -> "argN", assuming "arg0" is the
102  // canonical executable name. In the case that we didn't
103  // get enough tokens back from KERN_PROCARGS the following
104  // code will end up just keeping 'command' None (i.e.,
105  // tokens.size() will be <= 0).
106  tokens.erase(tokens.begin()); // Remove path.
107 
108  // In practice it appeared that we might get an 'argc'
109  // which is greater than 'tokens.size()' which caused us
110  // to get an invalid iterator from 'tokens.begin() +
111  // argc', thus we now check that condition here.
112  if (static_cast<size_t>(argc) <= tokens.size()) {
113  tokens.erase(tokens.begin() + argc, tokens.end());
114  }
115 
116  // If any tokens are left, consider them the command!
117  if (tokens.size() > 0) {
118  command = strings::join(" ", tokens);
119  }
120  }
121  }
122  }
123  }
124  }
125 #endif

127  // We also use proc_pidinfo() to get memory and CPU usage.
128  // NOTE: There are several pitfalls to using proc_pidinfo().
129  // In particular:
130  // -This will not work for many root processes.
131  // -This may not work for processes owned by other users.
132  // -However, this always works for processes owned by the same user.
133  // This beats using task_for_pid(), which only works for the same pid.
134  // For further discussion around these issues,
135  // see: http://code.google.com/p/psutil/issues/detail?id=297
136  proc_taskinfo task;
137  int size = proc_pidinfo(pid, PROC_PIDTASKINFO, 0, &task, sizeof(task));
138 
139  // It appears that zombie processes on OS X do not have sessions and
140  // result in ESRCH.
141  int session = getsid(pid);
142 
143  if (size != sizeof(task)) {
144  return Process(process.kp_proc.p_pid,
145  process.kp_eproc.e_ppid,
146  process.kp_eproc.e_pgid,
147  session > 0 ? session : Option<pid_t>::none(),
148  None(),
149  None(),
150  None(),
151  command.get(std::string(process.kp_proc.p_comm)),
152  process.kp_proc.p_stat & SZOMB);
153  } else {
154  return Process(process.kp_proc.p_pid,
155  process.kp_eproc.e_ppid,
156  process.kp_eproc.e_pgid,
157  session > 0 ? session : Option<pid_t>::none(),
158  Bytes(task.pti_resident_size),
159  Nanoseconds(task.pti_total_user),
160  Nanoseconds(task.pti_total_system),
161  command.get(std::string(process.kp_proc.p_comm)),
162  process.kp_proc.p_stat & SZOMB);
163  }
164 }

167 inline Try<std::set<pid_t> > pids()
168 {
169  const Try<int>& maxproc = os::sysctl(CTL_KERN, KERN_MAXPROC).integer();
170 
171  if (maxproc.isError()) {
172  return Error(maxproc.error());
173  }
174 
175  const Try<std::vector<kinfo_proc> >& processes =
176  os::sysctl(CTL_KERN, KERN_PROC, KERN_PROC_ALL).table(maxproc.get());
177 
178  if (processes.isError()) {
179  return Error(processes.error());
180  }
181 
182  std::set<pid_t> result;
183  foreach (const kinfo_proc& process, processes.get()) {
184  result.insert(process.kp_proc.p_pid);
185  }
186  return result;
187 }
