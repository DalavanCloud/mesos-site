527 template <class RoleSorter, class FrameworkSorter>
528 void
529 HierarchicalAllocatorProcess<RoleSorter, FrameworkSorter>::resourcesRecovered(
530  const FrameworkID& frameworkId,
531  const SlaveID& slaveId,
532  const Resources& resources,
533  const Option<Filters>& filters)
534 {
535  CHECK(initialized);
536 
537  if (resources.allocatable().size() == 0) {
538  return;
539  }
540 
541  // Updated resources allocated to framework (if framework still
542  // exists, which it might not in the event that we dispatched
543  // Master::offer before we received AllocatorProcess::frameworkRemoved
544  // or AllocatorProcess::frameworkDeactivated, in which case we will
545  // have already recovered all of its resources).
546  if (frameworks.contains(frameworkId) &&
547  sorters[frameworks[frameworkId].role()]->contains(frameworkId.value())) {
548  const std::string& role = frameworks[frameworkId].role();
549  sorters[role]->unallocated(frameworkId.value(), resources);
550  sorters[role]->remove(resources);
551  roleSorter->unallocated(role, resources);
552  }
553 
554  // Update resources allocatable on slave (if slave still exists,
555  // which it might not in the event that we dispatched Master::offer
556  // before we received Allocator::slaveRemoved).
557  if (slaves.contains(slaveId)) {
558  slaves[slaveId].available += resources;
559 
560  LOG(INFO) << "Recovered " << resources.allocatable()
561  << " (total allocatable: " << slaves[slaveId].available
562  << ") on slave " << slaveId
563  << " from framework " << frameworkId;
564  }
565 
566  // Create a filter for this slave/framework pair if both exist.
567  if (frameworks.contains(frameworkId) && slaves.contains(slaveId)) {
568  // Create a refused resources filter.
569  Try<Duration> seconds_ = Duration::create(Filters().refuse_seconds());
570  CHECK_SOME(seconds_);
571  Duration seconds = seconds_.get();
572 
573  // Update the value of 'seconds' if the input isSome() and is
574  // valid.
575  if (filters.isSome()) {
576  seconds_ = Duration::create(filters.get().refuse_seconds());
577  if (seconds_.isError()) {
578  LOG(WARNING) << "Using the default value of 'refuse_seconds' to create "
579  << "the refused resources filter because the input value "
580  << "is invalid: " << seconds_.error();
581  } else if (seconds_.get() < Duration::zero()) {
582  LOG(WARNING) << "Using the default value of 'refuse_seconds' to create "
583  << "the refused resources filter because the input value "
584  << "is negative";
585  } else {
586  seconds = seconds_.get();
587  }
588  }
589 
590  if (seconds != Duration::zero()) {
591  VLOG(1) << "Framework " << frameworkId
592  << " filtered slave " << slaveId
593  << " for " << seconds;
594 
595  // Create a new filter and delay its expiration.
596  Filter* filter =
597  new RefusedFilter(slaveId, resources, process::Timeout::in(seconds));
598 
599  frameworks[frameworkId].filters.insert(filter);
600 
601  delay(seconds, self(), &Self::expire, frameworkId, filter);
602  }
603  }
604 }

675 template <class RoleSorter, class FrameworkSorter>
676 void
677 HierarchicalAllocatorProcess<RoleSorter, FrameworkSorter>::allocate(
678  const hashset<SlaveID>& slaveIds)
679 {
680  CHECK(initialized);
681 
682  if (roleSorter->count() == 0) {
683  VLOG(1) << "No roles to allocate resources!";
684  return;
685  }
686 
687  if (slaveIds.empty()) {
688  VLOG(1) << "No resources available to allocate!";
689  return;
690  }
691 
692  foreach (const std::string& role, roleSorter->sort()) {
693  foreach (const std::string& frameworkIdValue, sorters[role]->sort()) {
694  FrameworkID frameworkId;
695  frameworkId.set_value(frameworkIdValue);
696 
697  Resources allocatedResources;
698  hashmap<SlaveID, Resources> offerable;
699  foreach (const SlaveID& slaveId, slaveIds) {
700  Resources unreserved = slaves[slaveId].available.extract("*");
701  Resources resources = unreserved;
702 
703  if (role != "*") {
704  resources += slaves[slaveId].available.extract(role);
705  }
706 
707  // Check whether or not this framework filters this slave.
708  bool filtered = isFiltered(frameworkId, slaveId, resources);
709 
710  if (!filtered &&
711  slaves[slaveId].activated &&
712  slaves[slaveId].whitelisted &&
713  allocatable(resources)) {
714  VLOG(1)
715  << "Offering " << resources << " on slave " << slaveId
716  << " to framework " << frameworkId;
717 
718  offerable[slaveId] = resources;
719 
720  // Update framework and slave resources.
721  slaves[slaveId].available -= resources;
722 
723  // We only count resources not reserved for this role
724  // in the share the sorter considers.
725  allocatedResources += unreserved;
726  }
727  }
728 
729  if (!offerable.empty()) {
730  sorters[role]->add(allocatedResources);
731  sorters[role]->allocated(frameworkIdValue, allocatedResources);
732  roleSorter->allocated(role, allocatedResources);
733 
734  dispatch(master, &Master::offer, frameworkId, offerable);
735  }
736  }
737  }
738 }

742 void
743 HierarchicalAllocatorProcess<RoleSorter, FrameworkSorter>::expire(
744  const FrameworkID& frameworkId,
745  Filter* filter)
746 {
747  // The filter might have already been removed (e.g., if the
748  // framework no longer exists or in
749  // HierarchicalAllocatorProcess::offersRevived) but not yet deleted (to
750  // keep the address from getting reused possibly causing premature
751  // expiration).
752  if (frameworks.contains(frameworkId) &&
753  frameworks[frameworkId].filters.contains(filter)) {
754  frameworks[frameworkId].filters.erase(filter);
755  }
756 
757  delete filter;
758 }

775 template <class RoleSorter, class FrameworkSorter>
776 bool
777 HierarchicalAllocatorProcess<RoleSorter, FrameworkSorter>::isFiltered(
778  const FrameworkID& frameworkId,
779  const SlaveID& slaveId,
780  const Resources& resources)
781 {
782  CHECK(frameworks.contains(frameworkId));
783  CHECK(slaves.contains(slaveId));
784 
785  // Do not offer a non-checkpointing slave's resources to a checkpointing
786  // framework. This is a short term fix until the following is resolved:
787  // https://issues.apache.org/jira/browse/MESOS-444.
788  if (frameworks[frameworkId].checkpoint && !slaves[slaveId].checkpoint) {
789  VLOG(1) << "Filtered " << resources
790  << " on non-checkpointing slave " << slaveId
791  << " for checkpointing framework " << frameworkId;
792  return true;
793  }
