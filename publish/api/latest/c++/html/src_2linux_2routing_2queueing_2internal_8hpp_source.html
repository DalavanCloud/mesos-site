72 // Encodes a queueing discipline (in our representation) to a libnl
73 // queueing discipline (rtnl_qdisc). We use template here so that it
74 // works for any type of queueing discipline.
75 template <typename Discipline>
76 Try<Netlink<struct rtnl_qdisc> > encode(
77  const Netlink<struct rtnl_link>& link,
78  const Discipline& discipline)
79 {
80  struct rtnl_qdisc* q = rtnl_qdisc_alloc();
81  if (q == NULL) {
82  return Error("Failed to allocate a libnl qdisc");
83  }
84 
85  Netlink<struct rtnl_qdisc> qdisc(q);
86 
87  rtnl_tc_set_link(TC_CAST(qdisc.get()), link.get());
88 
89  // Perform queue discipline specific encoding.
90  Try<Nothing> encoding = encode(qdisc, discipline);
91  if (encoding.isError()) {
92  return Error(
93  "Failed to encode the queueing discipline: " +
94  encoding.error());
95  }
96 
97  return qdisc;
98 }

104 // Returns all the libnl queue discipline (rtnl_qdisc) on the link.
105 inline Try<std::vector<Netlink<struct rtnl_qdisc> > > getQdiscs(
106  const Netlink<struct rtnl_link>& link)
107 {
108  Try<Netlink<struct nl_sock> > sock = routing::socket();
109  if (sock.isError()) {
110  return Error(sock.error());
111  }
112 
113  // Dump all the queueing discipline from kernel.
114  struct nl_cache* c = NULL;
115  int err = rtnl_qdisc_alloc_cache(sock.get().get(), &c);
116  if (err != 0) {
117  return Error(
118  "Failed to get queueing discipline info from kernel: " +
119  std::string(nl_geterror(err)));
120  }
121 
122  Netlink<struct nl_cache> cache(c);
123 
124  std::vector<Netlink<struct rtnl_qdisc> > results;
125 
126  for (struct nl_object* o = nl_cache_get_first(cache.get());
127  o != NULL; o = nl_cache_get_next(o)) {
128  nl_object_get(o); // Increment the reference counter.
129  results.push_back(Netlink<struct rtnl_qdisc>((struct rtnl_qdisc*) o));
130  }
131 
132  return results;
133 }

136 // Returns the libnl queueing discipline (rtnl_qdisc) that matches the
137 // specified queueing discipline on the link. Return None if no match
138 // has been found. We use template here so that it works for any type
139 // of queueing discipline.
140 template <typename Discipline>
141 Result<Netlink<struct rtnl_qdisc> > getQdisc(
142  const Netlink<struct rtnl_link>& link,
143  const Discipline& discipline)
144 {
145  Try<std::vector<Netlink<struct rtnl_qdisc> > > qdiscs = getQdiscs(link);
146  if (qdiscs.isError()) {
147  return Error(qdiscs.error());
148  }
149 
150  foreach (const Netlink<struct rtnl_qdisc>& qdisc, qdiscs.get()) {
151  // The decode function will return None if 'qdisc' does not match
152  // the specified queueing discipline. In that case, we just move
153  // on to the next libnl queueing discipline.
154  Result<Discipline> result = decode<Discipline>(qdisc);
155  if (result.isError()) {
156  return Error("Failed to decode: " + result.error());
157  } else if (result.isSome() && result.get() == discipline) {
158  return qdisc;
159  }
160  }
161 
162  return None();
163 }

190 // Creates a new queueing discipline on the link. Returns false if the
191 // same queueing discipline already exists on the link. We use
192 // template here so that it works for any type of queueing discipline.
193 template <typename Discipline>
194 Try<bool> create(const std::string& _link, const Discipline& discipline)
195 {
196  Result<Netlink<struct rtnl_link> > link = link::internal::get(_link);
197  if (link.isError()) {
198  return Error(link.error());
199  } else if (link.isNone()) {
200  return Error("Link '" + _link + "' is not found");
201  }
202 
203  Try<Netlink<struct rtnl_qdisc> > qdisc = encode(link.get(), discipline);
204  if (qdisc.isError()) {
205  return Error("Failed to encode the queueing discipline: " + qdisc.error());
206  }
207 
208  Try<Netlink<struct nl_sock> > sock = routing::socket();
209  if (sock.isError()) {
210  return Error(sock.error());
211  }
212 
213  // The flag NLM_F_EXCL tells libnl that if the qdisc already exists,
214  // this function should return error.
215  int err = rtnl_qdisc_add(
216  sock.get().get(),
217  qdisc.get().get(),
218  NLM_F_CREATE | NLM_F_EXCL);
219 
220  if (err != 0) {
221  if (err == -NLE_EXIST) {
222  return false;
223  }
224  return Error(
225  "Failed to add a queueing discipline to the link: " +
226  std::string(nl_geterror(err)));
227  }
228 
229  return true;
230 }
