28 // Provides a hash map via Boost's 'unordered_map'. For most intensive
29 // purposes this could be accomplished with a templated typedef, but
30 // those don't exist (until C++-11). Also, doing it this way allows us
31 // to add functionality, or better naming of existing functionality,
32 // etc.

69  {
70  typedef typename boost::unordered_map<Key, Value>::const_iterator
71  const_iterator;
72  const_iterator it = boost::unordered_map<Key, Value>::find(key);
73  if (it == boost::unordered_map<Key, Value>::end()) {
74  return None();
